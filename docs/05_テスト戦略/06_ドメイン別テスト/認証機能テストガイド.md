# 認証機能テストガイド

## 概要

本ガイドでは、Stats47プロジェクトの認証機能のテスト方法について説明します。NextAuth.js (Auth.js) v5を基盤とした認証システムのテスト戦略と実装例を提供します。

### 目的

1. **品質保証**: 認証機能の正確性を自動的に検証
2. **回帰テスト**: 変更による既存機能の破壊を防止
3. **セキュリティ**: 認証・認可の脆弱性を早期発見
4. **開発速度向上**: 自動化されたテストによる迅速なフィードバック

```typescript
import { describe, expect, it, vi } from "vitest";
```

```typescript
  signIn: vi.fn(),

      expect(mockSignIn).toHaveBeenCalledWith("credentials", {
        email: "test@example.com",
        password: "password123",
        redirect: false,
      });
    });
  });

  it("間違った認証情報でエラーを表示する", async () => {
    const user = userEvent.setup();
    const mockSignIn = vi.mocked(signIn);
    mockSignIn.mockResolvedValue({ 
      error: "CredentialsSignin", 
      ok: false, 
      status: 401, 
      url: null 
    });

    render(<LoginForm />);

    await user.type(screen.getByLabelText("メールアドレス"), "wrong@example.com");
    await user.type(screen.getByLabelText("パスワード"), "wrongpassword");
    await user.click(screen.getByRole("button", { name: "ログイン" }));

    await waitFor(() => {
      expect(screen.getByText(/認証に失敗しました/i)).toBeInTheDocument();
    });
  });

  it("バリデーションエラーを表示する", async () => {
    const user = userEvent.setup();

    render(<LoginForm />);

    await user.click(screen.getByRole("button", { name: "ログイン" }));

    await waitFor(() => {
      expect(screen.getByText(/メールアドレスは必須です/i)).toBeInTheDocument();
      expect(screen.getByText(/パスワードは必須です/i)).toBeInTheDocument();
    });
  });
});
```

#### AuthServiceのテスト例

```typescript
// src/features/auth/services/AuthService.test.ts
import { describe, expect, it, vi, beforeEach } from "vitest";

import { AuthService } from "./AuthService";
import { runQuery } from "../db/d1";
import mockUsersData from "@data/mock/auth/users.json";

// D1データベースのモック
vi.mock("../db/d1", () => ({
  runQuery: vi.fn(),
}));

describe("AuthService", () => {
  beforeEach(() => {
    vi.clearAllMocks();
  });

  describe("login", () => {
    it("モックデータで正しい認証情報でログインできる", async () => {
      // テスト用にモックデータを使用（環境変数ではなく、直接モックデータを参照）
      // 注意: 実際の実装では、モックデータはテスト用に直接インポート
      
      const result = await AuthService.login(
        "admin@example.com",
        "admin123"
      );

      expect(result).toBeDefined();
      expect(result?.email).toBe("admin@example.com");
    });

    it("モックデータで間違った認証情報でnullを返す", async () => {
      const result = await AuthService.login(
        "wrong@example.com",
        "wrongpassword"
      );

      expect(result).toBeNull();
    });

    it("D1データベースでユーザーを取得できる", async () => {
      const mockRunQuery = vi.mocked(runQuery);
      
      mockRunQuery.mockResolvedValue({
        results: [
          {
            id: "1",
            email: "test@example.com",
            username: "testuser",
            password_hash: "hashedpassword",
            role: "user",
            is_active: true,
          },
        ],
        success: true,
      });

      const result = await AuthService.login(
        "test@example.com",
        "hashedpassword"
      );

      expect(result).toBeDefined();
      expect(mockRunQuery).toHaveBeenCalledWith(
        "SELECT * FROM users WHERE email = ? AND password_hash = ?",
        "test@example.com",
        "hashedpassword"
      );
    });
  });

  describe("getCurrentUser", () => {
    it("モックデータでユーザーを取得できる", async () => {
      // テスト用にモックデータを使用
      const result = await AuthService.getCurrentUser("1");

      expect(result).toBeDefined();
      expect(result?.id).toBe("1");
    });

    it("存在しないユーザーIDでnullを返す", async () => {
      
      const result = await AuthService.getCurrentUser("999");

      expect(result).toBeNull();
    });
  });

  describe("checkPermission", () => {
    it("管理者ユーザーが管理者権限を持つ", () => {
      const adminUser = {
        id: "1",
        email: "admin@example.com",
        role: "admin",
      } as any;

      const result = AuthService.checkPermission(adminUser, "admin");
      expect(result).toBe(true);
    });

    it("一般ユーザーが管理者権限を持たない", () => {
      const normalUser = {
        id: "2",
        email: "user@example.com",
        role: "user",
      } as any;

      const result = AuthService.checkPermission(normalUser, "admin");
      expect(result).toBe(false);
    });

    it("nullユーザーでfalseを返す", () => {
      const result = AuthService.checkPermission(null, "admin");
      expect(result).toBe(false);
    });
  });
});
```

```typescript
import { render, screen, waitFor } from "@testing-library/react";
import userEvent from "@testing-library/user-event";

  });

    const user = userEvent.setup();
    const mockSignIn = vi.mocked(signIn);

    // 初期状態：未認証
    mockUseSession.mockReturnValue({
      data: null,
      status: "unauthenticated",
    });

    // ログイン処理

      error: null, 
      ok: true, 
      status: 200, 
      url: null 
    });

    await user.type(screen.getByLabelText("メールアドレス"), "test@example.com");
    await user.type(screen.getByLabelText("パスワード"), "password123");
    await user.click(screen.getByRole("button", { name: "ログイン" }));

    // セッション更新をシミュレート
    await waitFor(() => {
      mockUseSession.mockReturnValue({
        data: {
          user: {
            id: "1",
            email: "test@example.com",
            name: "テストユーザー",
            role: "user",
          },
        },
        status: "authenticated",
      });
    });
```

```typescript
      : null,

}

}
```

## ベストプラクティス

### 1. テストの独立性

各テストは独立して実行できるようにする：

```typescript
// ❌ 悪い例：テスト間で状態を共有
let session: Session | null = null;

describe("認証テスト", () => {
  it("ログイン", () => {
    session = await login();
  });

  it("セッション確認", () => {
    expect(session).toBeDefined(); // 前のテストに依存
  });
});

// ✅ 良い例：各テストで状態を初期化
describe("認証テスト", () => {
  it("ログイン", async () => {
    const session = await login();
    expect(session).toBeDefined();
  });

  it("セッション確認", async () => {
    const session = await getSession();
    expect(session).toBeDefined();
  });
});
```

### 2. モックの適切な使用

実際のAPI呼び出しを避け、モックを使用する：

```typescript
// ❌ 悪い例：実際のAPIを呼び出す
it("ログイン", async () => {
  const result = await signIn("credentials", {
    email: "test@example.com",
    password: "password",
  });
  // 実際のデータベースにアクセスしてしまう
});

// ✅ 良い例：モックを使用
it("ログイン", async () => {
  const mockSignIn = vi.mocked(signIn);
  mockSignIn.mockResolvedValue({ 
    error: null, 
    ok: true, 
    status: 200, 
    url: null 
  });

  const result = await signIn("credentials", {
    email: "test@example.com",
    password: "password",
  });
```

### 4. 非同期処理の適切な待機

`waitFor`を使用して非同期処理を待つ：

```typescript
// ❌ 悪い例：非同期処理を待たない
it("ログイン後の状態確認", async () => {
  await user.click(loginButton);
  expect(screen.getByText("ログイン成功")).toBeInTheDocument();
  // 状態更新を待たずにアサートしてしまう可能性がある
});

// ✅ 良い例：waitForを使用
it("ログイン後の状態確認", async () => {
  await user.click(loginButton);
  await waitFor(() => {
    expect(screen.getByText("ログイン成功")).toBeInTheDocument();
  });
});
```

```typescript
// vi.mockはテストファイルの最上部に配置
vi.mock("next-auth/react", () => ({

  signOut: vi.fn(),

beforeEach(() => {
  vi.clearAllMocks();
```

### Q2: 非同期処理のテストが不安定

**問題**: タイミングによってテストが失敗する

**解決策**:
```typescript
// waitForを使用して非同期処理を待つ
await waitFor(() => {
  expect(screen.getByText("期待されるテキスト")).toBeInTheDocument();
}, { timeout: 3000 });
```

```typescript
// テスト内で環境変数を設定
beforeEach(() => {
  process.env.NEXT_PUBLIC_ENV = "development";
```
