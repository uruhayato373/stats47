### データベースバックアップ

```typescript
// src/infrastructure/monitoring/database-health.ts
export async function checkDatabaseHealth(db: D1Database): Promise<{
  isHealthy: boolean;
  responseTime: number;
  error?: string;
  metrics?: {
    userCount: number;
    metainfoCount: number;
    lastUpdate: string;
  };
}> {
  const startTime = Date.now();

  try {
    // 基本的な接続テスト
    const healthCheck = await db.prepare("SELECT 1 as health_check").first();

    if (!healthCheck) {
      throw new Error("Health check query failed");
    }

    // メトリクス収集
    const userCount = await db
      .prepare("SELECT COUNT(*) as count FROM users")
      .first();
    const metainfoCount = await db
      .prepare("SELECT COUNT(*) as count FROM estat_metainfo")
      .first();
    const lastUpdate = await db
      .prepare("SELECT MAX(updated_at) as last_update FROM estat_metainfo")
      .first();

    const responseTime = Date.now() - startTime;

    return {
      isHealthy: true,
      responseTime,
      metrics: {
        userCount: userCount?.count || 0,
        metainfoCount: metainfoCount?.count || 0,
        lastUpdate: lastUpdate?.last_update || "N/A",
      },
    };
  } catch (error) {
    const responseTime = Date.now() - startTime;

    return {
      isHealthy: false,
      responseTime,
      error: error instanceof Error ? error.message : "Unknown error",
    };
  }
}
```

```typescript
  isHealthy: boolean;
  responseTime: number;
  error?: string;
  metrics?: {
    objectCount: number;
    totalSize: number;
    lastModified: string;
  };
}> {
  const startTime = Date.now();

  try {
    // 基本的なアクセステスト
    const testKey = "health-check/test.txt";
    const testData = new TextEncoder().encode("health check");

    // 書き込みテスト
    await bucket.put(testKey, testData);

    // 読み込みテスト
    const object = await bucket.get(testKey);
    if (!object) {
      throw new Error("Failed to read test object");
    }

    // 削除テスト
    await bucket.delete(testKey);

    // メトリクス収集
    const list = await bucket.list({ prefix: "estat_metainfo/", limit: 1000 });
    const objectCount = list.objects.length;
    const totalSize = list.objects.reduce(
      (sum, obj) => sum + (obj.size || 0),
      0
    );
    const lastModified =
      list.objects.length > 0
        ? list.objects[0].uploaded?.toISOString() || "N/A"
        : "N/A";

    const responseTime = Date.now() - startTime;

    return {
      isHealthy: true,
      responseTime,
      metrics: {
        objectCount,
        totalSize,
        lastModified,
      },
    };
  } catch (error) {
    const responseTime = Date.now() - startTime;

    return {
      isHealthy: false,
```

```typescript
// src/infrastructure/monitoring/query-performance.ts
export async function monitorQueryPerformance<T>(
  queryName: string,
  operation: () => Promise<T>
): Promise<T> {
  const startTime = Date.now();

  try {
    const result = await operation();
    const duration = Date.now() - startTime;

    // パフォーマンスログ
    logDatabaseOperation(queryName, "", duration, true);

    // スロークエリの警告
    if (duration > 1000) {
      console.warn(`Slow query detected: ${queryName} took ${duration}ms`);
    }

    return result;
  } catch (error) {
    const duration = Date.now() - startTime;
    logDatabaseOperation(queryName, "", duration, false, error.message);
    throw error;
  }
}

// 使用例
export async function getUsersWithMonitoring(db: D1Database) {
  return await monitorQueryPerformance("getUsers", async () => {
    const stmt = db.prepare(
      "SELECT * FROM users ORDER BY created_at DESC LIMIT 100"
    );
    return await stmt.all();
  });
}
```

```typescript
// src/infrastructure/monitoring/r2-performance.ts
export async function monitorR2Performance<T>(
  operationName: string,
  operation: () => Promise<T>
): Promise<T> {
  const startTime = Date.now();

  try {
    const result = await operation();
    const duration = Date.now() - startTime;

    // パフォーマンスログ
    logR2Operation(operationName as any, "", duration, true);

    // スローオペレーションの警告
    if (duration > 2000) {
      console.warn(
        `Slow R2 operation detected: ${operationName} took ${duration}ms`
      );
    }

    return result;
  } catch (error) {
    const duration = Date.now() - startTime;
    logR2Operation(
      operationName as any,
      "",
      duration,
      false,
      undefined,
      error.message
    );
    throw error;
  }
}

// 使用例
export async function saveMetaInfoWithMonitoring(
  bucket: R2Bucket,
  key: string,
  data: any
) {
  return await monitorR2Performance("saveMetaInfo", async () => {
    const jsonString = JSON.stringify(data, null, 2);
    const jsonBuffer = new TextEncoder().encode(jsonString);

    await bucket.put(key, jsonBuffer, {
      httpMetadata: {
        contentType: "application/json",
      },
    });

    return { key, size: jsonBuffer.byteLength };
  });
}
```

   ```typescript
   export async function parallelR2Read(bucket: R2Bucket, keys: string[]) {
     const results = await Promise.allSettled(
       keys.map(async (key) => {
         const object = await bucket.get(key);
         return object ? await object.json() : null;
       })
     );

     return results.map((result, index) => ({
       key: keys[index],
       data: result.status === "fulfilled" ? result.value : null,
       error: result.status === "rejected" ? result.reason : null,
     }));
   }
   ```

   ```typescript
   // メモリキャッシュの実装
   class R2Cache {
     private cache = new Map<string, { data: any; expires: number }>();
     private ttl = 5 * 60 * 1000; // 5分

     async get(key: string, bucket: R2Bucket): Promise<any> {
       const cached = this.cache.get(key);

       if (cached && cached.expires > Date.now()) {
         return cached.data;
       }

       const object = await bucket.get(key);
       const data = object ? await object.json() : null;

       if (data) {
         this.cache.set(key, {
           data,
           expires: Date.now() + this.ttl,
         });
       }

       return data;
     }
   }
   ```

```typescript
// src/infrastructure/maintenance/r2-cleanup.ts
export async function cleanupR2Storage(bucket: R2Bucket) {
  const cutoffDate = new Date();
  cutoffDate.setMonth(cutoffDate.getMonth() - 3); // 3ヶ月前

  // 古いファイルの一覧取得
  const list = await bucket.list({
    prefix: "temp/",
    limit: 1000,
  });

  let deletedCount = 0;
  const deletePromises = [];

  for (const object of list.objects) {
    if (object.uploaded && object.uploaded < cutoffDate) {
      deletePromises.push(bucket.delete(object.key));
      deletedCount++;
    }
  }

  await Promise.all(deletePromises);

  console.log(`Cleaned up ${deletedCount} old R2 objects`);

  return { deletedCount };
}
```

```typescript
// src/infrastructure/maintenance/r2-capacity-monitoring.ts
export async function monitorR2Capacity(bucket: R2Bucket) {
  const list = await bucket.list({ limit: 1000 });

  const totalSize = list.objects.reduce((sum, obj) => sum + (obj.size || 0), 0);
  const objectCount = list.objects.length;

  // サイズ別の分類
  const sizeCategories = {
    small: 0, // < 1MB
    medium: 0, // 1MB - 10MB
    large: 0, // > 10MB
  };

  for (const object of list.objects) {
    const size = object.size || 0;
    if (size < 1024 * 1024) {
      sizeCategories.small++;
    } else if (size < 10 * 1024 * 1024) {
      sizeCategories.medium++;
    } else {
      sizeCategories.large++;
    }
  }

  console.log("R2 Capacity:", {
    totalSize: Math.round(totalSize / 1024 / 1024) + "MB",
    objectCount,
    sizeCategories,
  });

  return {
    totalSize,
    objectCount,
    sizeCategories,
  };
}
```

1. **データベース環境分離**

   - 開発・ステージング・本番環境の完全分離
   - ローカル D1 とリモート D1 の自動切り替え

2. **R2 ストレージの統合**

   - e-Stat API レスポンスのキャッシュ
   - カスタムドメインによるアクセス最適化

3. **認証・認可の強化**
   - 管理画面へのアクセス制限
   - 開発ツールの分離

#### パフォーマンスの改善

1. **データベース最適化**

   - インデックスの追加
   - クエリの最適化
   - 接続プールの実装

2. **R2 ストレージ最適化**
   - 並列処理の実装
   - キャッシュ戦略の導入
   - CDN の活用

#### 運用性の向上

1. **監視・ログの強化**

   - ヘルスチェックの実装
   - パフォーマンス監視
   - エラーログの集約

2. **バックアップ・復旧**
   - 自動バックアップの設定
   - 復旧手順の整備
   - 災害復旧計画
