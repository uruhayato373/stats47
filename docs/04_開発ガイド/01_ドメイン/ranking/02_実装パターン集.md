### REST API型アダプター

```typescript
export class RestApiAdapter implements RankingDataAdapter {
  readonly sourceId = 'rest-api';
  readonly sourceName = 'REST API Source';

  private async fetchData(params: AdapterFetchParams): Promise<any> {
    const url = this.buildUrl(params);
    const response = await fetch(url, {
      headers: {
        'Authorization': `Bearer ${process.env.API_KEY}`,
        'Content-Type': 'application/json',
      },
    });

    if (!response.ok) {
      throw new Error(`API request failed: ${response.status}`);
    }

    return response.json();
  }

  private buildUrl(params: AdapterFetchParams): string {
    const { stats_data_id, cd_cat01 } = params.sourceSpecific;
    return `https://api.example.com/v1/data?id=${stats_data_id}&category=${cd_cat01}`;
  }
}
```

### CSV/ファイル型アダプター

```typescript
export class CsvAdapter implements RankingDataAdapter {
  readonly sourceId = 'csv';
  readonly sourceName = 'CSV File';

  private async fetchData(params: AdapterFetchParams): Promise<any> {
    const { file_path } = params.sourceSpecific;

    // R2からCSVファイルを取得
    const r2Object = await this.r2.get(file_path);
    if (!r2Object) {
      throw new Error(`CSV file not found: ${file_path}`);
    }

    const csvContent = await r2Object.text();
    return this.parseCSV(csvContent);
  }

  private parseCSV(content: string): any[] {
    const lines = content.split('\n');
    const headers = lines[0].split(',');

    return lines.slice(1).map(line => {
      const values = line.split(',');
      return headers.reduce((obj, header, index) => {
        obj[header.trim()] = values[index]?.trim();
        return obj;
      }, {});
    });
  }
}
```

### API Routeでの取得

```typescript
// app/api/rankings/[key]/route.ts
import { NextRequest, NextResponse } from 'next/server';
import { RankingService } from '@/features/ranking/services/ranking-service';

export async function GET(
  request: NextRequest,
  { params }: { params: { key: string } }
) {
  const { searchParams } = new URL(request.url);
  const year = searchParams.get('year') || '2023';

  try {
    const service = await RankingService.create();
    const data = await service.getRankingData(params.key, year);

    return NextResponse.json(data);
  } catch (error) {
    return NextResponse.json(
      { error: error.message },
      { status: 500 }
    );
  }
}
```

## エラーハンドリングパターン

### アダプター内でのエラーハンドリング

```typescript
export class RobustAdapter implements RankingDataAdapter {
  async fetchAndTransform(params: AdapterFetchParams): Promise<UnifiedRankingData> {
    try {
      const rawData = await this.fetchData(params);
      return this.transformData(rawData);
    } catch (error) {
      if (error instanceof NetworkError) {
        // ネットワークエラー: リトライ
        return this.retryFetch(params);
      } else if (error instanceof ValidationError) {
        // バリデーションエラー: ログ記録して再スロー
        console.error('[Adapter] Validation error:', error);
        throw error;
      } else {
        // その他のエラー
        console.error('[Adapter] Unexpected error:', error);
        throw new Error(`Failed to fetch data: ${error.message}`);
      }
    }
  }

  private async retryFetch(
    params: AdapterFetchParams,
    retries = 3
  ): Promise<UnifiedRankingData> {
    for (let i = 0; i < retries; i++) {
      try {
        const rawData = await this.fetchData(params);
        return this.transformData(rawData);
      } catch (error) {
        if (i === retries - 1) throw error;
        await this.delay(1000 * (i + 1)); // 指数バックオフ
      }
    }
    throw new Error('Max retries exceeded');
  }

  private delay(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}
```

### Serviceレイヤーでのエラーハンドリング

```typescript
export class RankingService {
  async getRankingData(rankingKey: string, timeCode: string): Promise<RankingDataPoint[]> {
    try {
      // 1. キャッシュチェック
      const cached = await this.cacheService.get(rankingKey, timeCode);
      if (cached) return cached;

      // 2. データ取得
      const data = await this.fetchFromAdapter(rankingKey, timeCode);

      // 3. キャッシュ保存
      await this.cacheService.save(rankingKey, timeCode, data);

      return data.values;
    } catch (error) {
      console.error('[RankingService] Error:', error);

      // フォールバック: 古いキャッシュを返す
      const fallback = await this.cacheService.getLastValid(rankingKey);
      if (fallback) {
        console.warn('[RankingService] Using fallback cache');
        return fallback;
      }

      throw error;
    }
  }
}
```

### 1. TypeScript型の活用

```typescript
// 型安全なパラメータ定義
interface EstatParams {
  stats_data_id: string;
  cd_cat01: string;
  cd_area?: string;
}

// 型ガードで安全に抽出
function extractEstatParams(sourceSpecific: unknown): EstatParams {
  if (
    typeof sourceSpecific !== 'object' ||
    !sourceSpecific ||
    !('stats_data_id' in sourceSpecific) ||
    !('cd_cat01' in sourceSpecific)
  ) {
    throw new Error('Invalid e-Stat parameters');
  }

  return sourceSpecific as EstatParams;
}
```
