# DataIntegration ドメイン - エンティティ

## 概要

DataIntegration ドメインの主要エンティティを定義します。これらのエンティティは、外部データソースとの統合とキャッシュ管理の中核を担います。

## 主要エンティティ

### 1. DataSource（データソース）

外部データソースの情報を管理するエンティティ。

```typescript
export class DataSource {
  private constructor(
    private readonly _id: DataSourceId,
    private readonly _name: string,
    private readonly _type: DataSourceType,
    private readonly _endpoint: string,
    private readonly _authentication: AuthenticationConfig,
    private readonly _rateLimit: RateLimitConfig,
    private readonly _isActive: boolean = true,
    private readonly _createdAt: Date = new Date(),
    private readonly _updatedAt: Date = new Date()
  ) {}

  static create(props: DataSourceProps): DataSource {
    this.validateProps(props);
    return new DataSource(
      props.id,
      props.name,
      props.type,
      props.endpoint,
      props.authentication,
      props.rateLimit,
      props.isActive ?? true,
      props.createdAt ?? new Date(),
      props.updatedAt ?? new Date()
    );
  }

  // ゲッター
  get id(): DataSourceId {
    return this._id;
  }
  get name(): string {
    return this._name;
  }
  get type(): DataSourceType {
    return this._type;
  }
  get endpoint(): string {
    return this._endpoint;
  }
  get authentication(): AuthenticationConfig {
    return this._authentication;
  }
  get rateLimit(): RateLimitConfig {
    return this._rateLimit;
  }
  get isActive(): boolean {
    return this._isActive;
  }
  get createdAt(): Date {
    return this._createdAt;
  }
  get updatedAt(): Date {
    return this._updatedAt;
  }

  // ビジネスメソッド
  activate(): DataSource {
    return new DataSource(
      this._id,
      this._name,
      this._type,
      this._endpoint,
      this._authentication,
      this._rateLimit,
      true,
      this._createdAt,
      new Date()
    );
  }

  deactivate(): DataSource {
    return new DataSource(
      this._id,
      this._name,
      this._type,
      this._endpoint,
      this._authentication,
      this._rateLimit,
      false,
      this._createdAt,
      new Date()
    );
  }

  updateEndpoint(newEndpoint: string): DataSource {
    this.validateEndpoint(newEndpoint);
    return new DataSource(
      this._id,
      this._name,
      this._type,
      newEndpoint,
      this._authentication,
      this._rateLimit,
      this._isActive,
      this._createdAt,
      new Date()
    );
  }

  // プライベートメソッド
  private static validateProps(props: DataSourceProps): void {
    if (!props.id) throw new Error("DataSource ID is required");
    if (!props.name?.trim()) throw new Error("DataSource name is required");
    if (!props.endpoint?.trim())
      throw new Error("DataSource endpoint is required");
  }

  private validateEndpoint(endpoint: string): void {
    try {
      new URL(endpoint);
    } catch {
      throw new Error("Invalid endpoint URL");
    }
  }
}

// 値オブジェクト
export class DataSourceId {
  private constructor(private readonly _value: string) {}

  static create(value: string): DataSourceId {
    if (!value?.trim()) {
      throw new Error("DataSource ID cannot be empty");
    }
    return new DataSourceId(value);
  }

  get value(): string {
    return this._value;
  }

  equals(other: DataSourceId): boolean {
    return this._value === other._value;
  }
}

export enum DataSourceType {
  ESTAT_API = "estat_api",
  GEOSHAPE_API = "geoshape_api",
  WORLD_BANK_API = "world_bank_api",
  OECD_API = "oecd_api",
}

export interface AuthenticationConfig {
  type: "api_key" | "oauth" | "basic" | "none";
  credentials?: Record<string, string>;
}

export interface RateLimitConfig {
  requestsPerMinute: number;
  requestsPerDay: number;
  burstLimit: number;
}
```

### 2. CacheEntry（キャッシュエントリ）

R2 ストレージに保存されるキャッシュエントリを管理するエンティティ。

```typescript
export class CacheEntry {
  private constructor(
    private readonly _key: CacheKey,
    private readonly _data: any,
    private readonly _ttl: number,
    private readonly _metadata: CacheMetadata,
    private readonly _createdAt: Date = new Date(),
    private readonly _lastAccessedAt: Date = new Date(),
    private readonly _hitCount: number = 0
  ) {}

  static create(props: CacheEntryProps): CacheEntry {
    this.validateProps(props);
    return new CacheEntry(
      props.key,
      props.data,
      props.ttl,
      props.metadata,
      props.createdAt ?? new Date(),
      props.lastAccessedAt ?? new Date(),
      props.hitCount ?? 0
    );
  }

  // ゲッター
  get key(): CacheKey {
    return this._key;
  }
  get data(): any {
    return this._data;
  }
  get ttl(): number {
    return this._ttl;
  }
  get metadata(): CacheMetadata {
    return this._metadata;
  }
  get createdAt(): Date {
    return this._createdAt;
  }
  get lastAccessedAt(): Date {
    return this._lastAccessedAt;
  }
  get hitCount(): number {
    return this._hitCount;
  }

  // ビジネスメソッド
  isExpired(): boolean {
    const now = new Date();
    const expiresAt = new Date(this._createdAt.getTime() + this._ttl * 1000);
    return now > expiresAt;
  }

  recordHit(): CacheEntry {
    return new CacheEntry(
      this._key,
      this._data,
      this._ttl,
      this._metadata,
      this._createdAt,
      new Date(),
      this._hitCount + 1
    );
  }

  updateData(newData: any): CacheEntry {
    return new CacheEntry(
      this._key,
      newData,
      this._ttl,
      this._metadata,
      this._createdAt,
      this._lastAccessedAt,
      this._hitCount
    );
  }

  extendTTL(additionalSeconds: number): CacheEntry {
    return new CacheEntry(
      this._key,
      this._data,
      this._ttl + additionalSeconds,
      this._metadata,
      this._createdAt,
      this._lastAccessedAt,
      this._hitCount
    );
  }

  // プライベートメソッド
  private static validateProps(props: CacheEntryProps): void {
    if (!props.key) throw new Error("Cache key is required");
    if (props.data === undefined || props.data === null) {
      throw new Error("Cache data is required");
    }
    if (props.ttl <= 0) throw new Error("TTL must be positive");
  }
}

// 値オブジェクト
export class CacheKey {
  private constructor(private readonly _value: string) {}

  static create(value: string): CacheKey {
    if (!value?.trim()) {
      throw new Error("Cache key cannot be empty");
    }
    if (value.length > 255) {
      throw new Error("Cache key too long");
    }
    return new CacheKey(value);
  }

  get value(): string {
    return this._value;
  }

  equals(other: CacheKey): boolean {
    return this._value === other._value;
  }
}

export interface CacheMetadata {
  apiType: string;
  parameters: Record<string, any>;
  size: number;
  source: string;
  version: string;
}
```

### 3. ApiParameter（API パラメータ）

外部 API 呼び出しに使用するパラメータを管理するエンティティ。

```typescript
export class ApiParameter {
  private constructor(
    private readonly _id: ApiParameterId,
    private readonly _apiType: ApiType,
    private readonly _parameters: Record<string, any>,
    private readonly _isValid: boolean = true,
    private readonly _createdAt: Date = new Date()
  ) {}

  static create(props: ApiParameterProps): ApiParameter {
    this.validateProps(props);
    return new ApiParameter(
      props.id,
      props.apiType,
      props.parameters,
      props.isValid ?? true,
      props.createdAt ?? new Date()
    );
  }

  // ゲッター
  get id(): ApiParameterId {
    return this._id;
  }
  get apiType(): ApiType {
    return this._apiType;
  }
  get parameters(): Record<string, any> {
    return this._parameters;
  }
  get isValid(): boolean {
    return this._isValid;
  }
  get createdAt(): Date {
    return this._createdAt;
  }

  // ビジネスメソッド
  validate(): ApiParameter {
    const validationResult = this.validateParameters();
    return new ApiParameter(
      this._id,
      this._apiType,
      this._parameters,
      validationResult.isValid,
      this._createdAt
    );
  }

  updateParameter(key: string, value: any): ApiParameter {
    const newParameters = { ...this._parameters, [key]: value };
    return new ApiParameter(
      this._id,
      this._apiType,
      newParameters,
      this._isValid,
      this._createdAt
    );
  }

  removeParameter(key: string): ApiParameter {
    const newParameters = { ...this._parameters };
    delete newParameters[key];
    return new ApiParameter(
      this._id,
      this._apiType,
      newParameters,
      this._isValid,
      this._createdAt
    );
  }

  generateCacheKey(): CacheKey {
    const paramString = JSON.stringify(this._parameters);
    const hash = crypto
      .createHash("sha256")
      .update(paramString)
      .digest("hex")
      .substring(0, 12);

    return CacheKey.create(`${this._apiType}:${hash}`);
  }

  // プライベートメソッド
  private static validateProps(props: ApiParameterProps): void {
    if (!props.id) throw new Error("API parameter ID is required");
    if (!props.apiType) throw new Error("API type is required");
    if (!props.parameters || typeof props.parameters !== "object") {
      throw new Error("Parameters must be an object");
    }
  }

  private validateParameters(): { isValid: boolean; errors: string[] } {
    const errors: string[] = [];

    // API種別ごとのバリデーション
    switch (this._apiType) {
      case ApiType.GET_META_INFO:
        if (!this._parameters.statsDataId) {
          errors.push("statsDataId is required for getMetaInfo");
        }
        break;
      case ApiType.GET_STATS_DATA:
        if (!this._parameters.statsDataId) {
          errors.push("statsDataId is required for getStatsData");
        }
        if (!this._parameters.cdCat01) {
          errors.push("cdCat01 is required for getStatsData");
        }
        break;
      case ApiType.GET_STATS_LIST:
        if (!this._parameters.cdCat01) {
          errors.push("cdCat01 is required for getStatsList");
        }
        break;
    }

    return {
      isValid: errors.length === 0,
      errors,
    };
  }
}

// 値オブジェクト
export class ApiParameterId {
  private constructor(private readonly _value: string) {}

  static create(value: string): ApiParameterId {
    if (!value?.trim()) {
      throw new Error("API parameter ID cannot be empty");
    }
    return new ApiParameterId(value);
  }

  get value(): string {
    return this._value;
  }

  equals(other: ApiParameterId): boolean {
    return this._value === other._value;
  }
}

export enum ApiType {
  GET_META_INFO = "getMetaInfo",
  GET_STATS_DATA = "getStatsData",
  GET_STATS_LIST = "getStatsList",
}
```

### 4. CacheStatistics（キャッシュ統計）

キャッシュの使用状況とパフォーマンスを管理するエンティティ。

```typescript
export class CacheStatistics {
  private constructor(
    private readonly _id: CacheStatisticsId,
    private readonly _totalRequests: number,
    private readonly _hitCount: number,
    private readonly _missCount: number,
    private readonly _cacheSize: number,
    private readonly _averageResponseTime: number,
    private readonly _lastUpdated: Date = new Date()
  ) {}

  static create(props: CacheStatisticsProps): CacheStatistics {
    this.validateProps(props);
    return new CacheStatistics(
      props.id,
      props.totalRequests,
      props.hitCount,
      props.missCount,
      props.cacheSize,
      props.averageResponseTime,
      props.lastUpdated ?? new Date()
    );
  }

  // ゲッター
  get id(): CacheStatisticsId {
    return this._id;
  }
  get totalRequests(): number {
    return this._totalRequests;
  }
  get hitCount(): number {
    return this._hitCount;
  }
  get missCount(): number {
    return this._missCount;
  }
  get cacheSize(): number {
    return this._cacheSize;
  }
  get averageResponseTime(): number {
    return this._averageResponseTime;
  }
  get lastUpdated(): Date {
    return this._lastUpdated;
  }

  // 計算プロパティ
  get hitRate(): number {
    return this._totalRequests > 0 ? this._hitCount / this._totalRequests : 0;
  }

  get missRate(): number {
    return this._totalRequests > 0 ? this._missCount / this._totalRequests : 0;
  }

  // ビジネスメソッド
  recordHit(responseTime: number): CacheStatistics {
    const newTotalRequests = this._totalRequests + 1;
    const newHitCount = this._hitCount + 1;
    const newAverageResponseTime = this.calculateNewAverage(responseTime);

    return new CacheStatistics(
      this._id,
      newTotalRequests,
      newHitCount,
      this._missCount,
      this._cacheSize,
      newAverageResponseTime,
      new Date()
    );
  }

  recordMiss(responseTime: number): CacheStatistics {
    const newTotalRequests = this._totalRequests + 1;
    const newMissCount = this._missCount + 1;
    const newAverageResponseTime = this.calculateNewAverage(responseTime);

    return new CacheStatistics(
      this._id,
      newTotalRequests,
      this._hitCount,
      newMissCount,
      this._cacheSize,
      newAverageResponseTime,
      new Date()
    );
  }

  updateCacheSize(newSize: number): CacheStatistics {
    return new CacheStatistics(
      this._id,
      this._totalRequests,
      this._hitCount,
      this._missCount,
      newSize,
      this._averageResponseTime,
      new Date()
    );
  }

  reset(): CacheStatistics {
    return new CacheStatistics(this._id, 0, 0, 0, 0, 0, new Date());
  }

  // プライベートメソッド
  private static validateProps(props: CacheStatisticsProps): void {
    if (!props.id) throw new Error("Cache statistics ID is required");
    if (props.totalRequests < 0)
      throw new Error("Total requests cannot be negative");
    if (props.hitCount < 0) throw new Error("Hit count cannot be negative");
    if (props.missCount < 0) throw new Error("Miss count cannot be negative");
    if (props.cacheSize < 0) throw new Error("Cache size cannot be negative");
    if (props.averageResponseTime < 0)
      throw new Error("Average response time cannot be negative");
  }

  private calculateNewAverage(newResponseTime: number): number {
    const totalTime =
      this._averageResponseTime * this._totalRequests + newResponseTime;
    return totalTime / (this._totalRequests + 1);
  }
}

// 値オブジェクト
export class CacheStatisticsId {
  private constructor(private readonly _value: string) {}

  static create(value: string): CacheStatisticsId {
    if (!value?.trim()) {
      throw new Error("Cache statistics ID cannot be empty");
    }
    return new CacheStatisticsId(value);
  }

  get value(): string {
    return this._value;
  }

  equals(other: CacheStatisticsId): boolean {
    return this._value === other._value;
  }
}
```

## ビジネスルール

### 1. DataSource のルール

- データソース ID は一意でなければならない
- エンドポイント URL は有効な形式でなければならない
- レート制限は正の値でなければならない
- 非アクティブなデータソースは使用できない

### 2. CacheEntry のルール

- キャッシュキーは一意でなければならない
- TTL は正の値でなければならない
- 期限切れのキャッシュエントリは使用できない
- ヒット回数は負の値にならない

### 3. ApiParameter のルール

- API 種別に応じた必須パラメータが存在する
- パラメータの値は適切な型でなければならない
- 無効なパラメータは API 呼び出しに使用できない

### 4. CacheStatistics のルール

- ヒット数とミス数の合計は総リクエスト数と一致する
- ヒット率は 0 から 1 の間の値である
- 平均応答時間は負の値にならない
