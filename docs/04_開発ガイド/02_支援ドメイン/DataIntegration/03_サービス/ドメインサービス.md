# DataIntegration ドメイン - ドメインサービス

## 概要

DataIntegration ドメインのドメインサービスを定義します。これらのサービスは、複数のエンティティにまたがるビジネスロジックや、外部システムとの統合を担当します。

## 主要ドメインサービス

### 1. EstatCacheService

e-Stat API レスポンスのキャッシュ管理を担当するドメインサービス。

```typescript
export class EstatCacheService {
  constructor(
    private readonly cacheRepository: CacheRepository,
    private readonly estatApiRepository: EstatApiRepository,
    private readonly metadataRepository: MetadataRepository,
    private readonly statisticsService: CacheStatisticsService
  ) {}

  async getCachedResponse(
    apiType: ApiType,
    parameters: Record<string, any>
  ): Promise<any> {
    const startTime = Date.now();

    try {
      // キャッシュキーの生成
      const apiParameter = ApiParameter.create({
        id: ApiParameterId.create(crypto.randomUUID()),
        apiType,
        parameters,
      });

      const cacheKey = apiParameter.generateCacheKey();

      // キャッシュの確認
      const cachedEntry = await this.cacheRepository.findByKey(cacheKey);

      if (cachedEntry && !cachedEntry.isExpired()) {
        // キャッシュヒット
        const updatedEntry = cachedEntry.recordHit();
        await this.cacheRepository.save(updatedEntry);

        const responseTime = Date.now() - startTime;
        await this.statisticsService.recordHit(responseTime);

        return cachedEntry.data;
      }

      // キャッシュミス - API呼び出し
      const response = await this.callEstatApi(apiType, parameters);

      // キャッシュに保存
      const cacheEntry = CacheEntry.create({
        key: cacheKey,
        data: response,
        ttl: this.getTtlForApiType(apiType),
        metadata: {
          apiType: apiType,
          parameters: parameters,
          size: JSON.stringify(response).length,
          source: "estat_api",
          version: "1.0",
        },
      });

      await this.cacheRepository.save(cacheEntry);
      await this.metadataRepository.saveCacheMetadata(
        cacheKey,
        apiType,
        parameters
      );

      const responseTime = Date.now() - startTime;
      await this.statisticsService.recordMiss(responseTime);

      return response;
    } catch (error) {
      const responseTime = Date.now() - startTime;
      await this.statisticsService.recordMiss(responseTime);
      throw new EstatApiError(
        `Failed to get cached response: ${error.message}`,
        error
      );
    }
  }

  async invalidateByPattern(pattern: string): Promise<void> {
    try {
      const entries = await this.cacheRepository.findByPattern(pattern);

      for (const entry of entries) {
        await this.cacheRepository.delete(entry.key);
      }

      await this.metadataRepository.deleteByPattern(pattern);
    } catch (error) {
      throw new CacheError(
        `Failed to invalidate cache by pattern: ${error.message}`,
        error
      );
    }
  }

  async invalidateByApiType(apiType: ApiType): Promise<void> {
    const pattern = `estat:${apiType}:`;
    await this.invalidateByPattern(pattern);
  }

  async invalidateExpired(): Promise<void> {
    try {
      const expiredEntries = await this.cacheRepository.findExpired();

      for (const entry of expiredEntries) {
        await this.cacheRepository.delete(entry.key);
      }

      await this.metadataRepository.deleteExpired();
    } catch (error) {
      throw new CacheError(
        `Failed to invalidate expired cache: ${error.message}`,
        error
      );
    }
  }

  private async callEstatApi(
    apiType: ApiType,
    parameters: Record<string, any>
  ): Promise<any> {
    switch (apiType) {
      case ApiType.GET_META_INFO:
        return await this.estatApiRepository.getMetaInfo(parameters);
      case ApiType.GET_STATS_DATA:
        return await this.estatApiRepository.getStatsData(parameters);
      case ApiType.GET_STATS_LIST:
        return await this.estatApiRepository.getStatsList(parameters);
      default:
        throw new Error(`Unsupported API type: ${apiType}`);
    }
  }

  private getTtlForApiType(apiType: ApiType): number {
    const ttlMap = {
      [ApiType.GET_META_INFO]: 7 * 24 * 60 * 60, // 7日
      [ApiType.GET_STATS_DATA]: 24 * 60 * 60, // 24時間
      [ApiType.GET_STATS_LIST]: 7 * 24 * 60 * 60, // 7日
    };
    return ttlMap[apiType] || 24 * 60 * 60;
  }
}
```

### 2. GeoshapeCacheService

地理データ（TopoJSON）のキャッシュ管理を担当するドメインサービス。

```typescript
export class GeoshapeCacheService {
  constructor(
    private readonly cacheRepository: CacheRepository,
    private readonly geoshapeApiRepository: GeoshapeApiRepository,
    private readonly statisticsService: CacheStatisticsService
  ) {}

  async getCachedTopoJson(
    year: number,
    level: "prefecture" | "municipality",
    resolution: "low" | "medium" | "high" | "full" = "low"
  ): Promise<any> {
    const startTime = Date.now();

    try {
      // キャッシュキーの生成
      const cacheKey = CacheKey.create(
        `geo:topojson:${year}:${level}:${resolution}`
      );

      // キャッシュの確認
      const cachedEntry = await this.cacheRepository.findByKey(cacheKey);

      if (cachedEntry && !cachedEntry.isExpired()) {
        // キャッシュヒット
        const updatedEntry = cachedEntry.recordHit();
        await this.cacheRepository.save(updatedEntry);

        const responseTime = Date.now() - startTime;
        await this.statisticsService.recordHit(responseTime);

        return cachedEntry.data;
      }

      // キャッシュミス - Geoshape API呼び出し
      const topoJson = await this.geoshapeApiRepository.getTopoJson(
        year,
        level,
        resolution
      );

      // キャッシュに保存
      const cacheEntry = CacheEntry.create({
        key: cacheKey,
        data: topoJson,
        ttl: 7 * 24 * 60 * 60, // 7日
        metadata: {
          apiType: "geoshape_topojson",
          parameters: { year, level, resolution },
          size: JSON.stringify(topoJson).length,
          source: "geoshape_api",
          version: "1.0",
        },
      });

      await this.cacheRepository.save(cacheEntry);

      const responseTime = Date.now() - startTime;
      await this.statisticsService.recordMiss(responseTime);

      return topoJson;
    } catch (error) {
      const responseTime = Date.now() - startTime;
      await this.statisticsService.recordMiss(responseTime);
      throw new GeoshapeApiError(
        `Failed to get cached TopoJSON: ${error.message}`,
        error
      );
    }
  }

  async invalidateByYear(year: number): Promise<void> {
    const pattern = `geo:topojson:${year}:`;
    await this.cacheRepository.deleteByPattern(pattern);
  }

  async invalidateByLevel(level: "prefecture" | "municipality"): Promise<void> {
    const pattern = `geo:topojson:*:${level}:`;
    await this.cacheRepository.deleteByPattern(pattern);
  }
}
```

### 3. CacheInvalidationService

キャッシュの無効化を管理するドメインサービス。

```typescript
export class CacheInvalidationService {
  constructor(
    private readonly cacheRepository: CacheRepository,
    private readonly metadataRepository: MetadataRepository,
    private readonly statisticsService: CacheStatisticsService
  ) {}

  async invalidateByPattern(pattern: string): Promise<InvalidationResult> {
    const startTime = Date.now();

    try {
      // R2のキー一覧を取得
      const entries = await this.cacheRepository.findByPattern(pattern);

      if (entries.length === 0) {
        return {
          success: true,
          deletedCount: 0,
          duration: Date.now() - startTime,
        };
      }

      // バッチで削除
      const deletePromises = entries.map((entry) =>
        this.cacheRepository.delete(entry.key)
      );

      await Promise.all(deletePromises);

      // D1のメタデータも削除
      await this.metadataRepository.deleteByPattern(pattern);

      const duration = Date.now() - startTime;

      return {
        success: true,
        deletedCount: entries.length,
        duration,
      };
    } catch (error) {
      throw new CacheError(
        `Failed to invalidate cache by pattern: ${error.message}`,
        error
      );
    }
  }

  async invalidateByApiType(apiType: ApiType): Promise<InvalidationResult> {
    const pattern = `estat:${apiType}:`;
    return await this.invalidateByPattern(pattern);
  }

  async invalidateExpired(): Promise<InvalidationResult> {
    const startTime = Date.now();

    try {
      const expiredEntries = await this.cacheRepository.findExpired();

      if (expiredEntries.length === 0) {
        return {
          success: true,
          deletedCount: 0,
          duration: Date.now() - startTime,
        };
      }

      // 期限切れエントリを削除
      const deletePromises = expiredEntries.map((entry) =>
        this.cacheRepository.delete(entry.key)
      );

      await Promise.all(deletePromises);

      // メタデータも削除
      await this.metadataRepository.deleteExpired();

      const duration = Date.now() - startTime;

      return {
        success: true,
        deletedCount: expiredEntries.length,
        duration,
      };
    } catch (error) {
      throw new CacheError(
        `Failed to invalidate expired cache: ${error.message}`,
        error
      );
    }
  }

  async invalidateByAge(maxAgeHours: number): Promise<InvalidationResult> {
    const startTime = Date.now();
    const cutoffTime = new Date(Date.now() - maxAgeHours * 60 * 60 * 1000);

    try {
      const oldEntries = await this.cacheRepository.findOlderThan(cutoffTime);

      if (oldEntries.length === 0) {
        return {
          success: true,
          deletedCount: 0,
          duration: Date.now() - startTime,
        };
      }

      // 古いエントリを削除
      const deletePromises = oldEntries.map((entry) =>
        this.cacheRepository.delete(entry.key)
      );

      await Promise.all(deletePromises);

      const duration = Date.now() - startTime;

      return {
        success: true,
        deletedCount: oldEntries.length,
        duration,
      };
    } catch (error) {
      throw new CacheError(
        `Failed to invalidate cache by age: ${error.message}`,
        error
      );
    }
  }

  async invalidateAll(): Promise<InvalidationResult> {
    const startTime = Date.now();

    try {
      const allEntries = await this.cacheRepository.findAll();

      if (allEntries.length === 0) {
        return {
          success: true,
          deletedCount: 0,
          duration: Date.now() - startTime,
        };
      }

      // 全エントリを削除
      const deletePromises = allEntries.map((entry) =>
        this.cacheRepository.delete(entry.key)
      );

      await Promise.all(deletePromises);

      // 全メタデータを削除
      await this.metadataRepository.deleteAll();

      const duration = Date.now() - startTime;

      return {
        success: true,
        deletedCount: allEntries.length,
        duration,
      };
    } catch (error) {
      throw new CacheError(
        `Failed to invalidate all cache: ${error.message}`,
        error
      );
    }
  }
}

export interface InvalidationResult {
  success: boolean;
  deletedCount: number;
  duration: number;
}
```

### 4. CacheStatsService

キャッシュ統計情報の取得と管理を担当するドメインサービス。

```typescript
export class CacheStatsService {
  constructor(
    private readonly statisticsRepository: StatisticsRepository,
    private readonly cacheRepository: CacheRepository
  ) {}

  async getStatistics(): Promise<CacheStatistics> {
    try {
      const stats = await this.statisticsRepository.findLatest();

      if (!stats) {
        // 初期統計を作成
        return CacheStatistics.create({
          id: CacheStatisticsId.create("default"),
          totalRequests: 0,
          hitCount: 0,
          missCount: 0,
          cacheSize: 0,
          averageResponseTime: 0,
        });
      }

      return stats;
    } catch (error) {
      throw new StatisticsError(
        `Failed to get cache statistics: ${error.message}`,
        error
      );
    }
  }

  async getStatisticsByApiType(apiType: ApiType): Promise<CacheStatistics> {
    try {
      const stats = await this.statisticsRepository.findByApiType(apiType);

      if (!stats) {
        return CacheStatistics.create({
          id: CacheStatisticsId.create(`api_${apiType}`),
          totalRequests: 0,
          hitCount: 0,
          missCount: 0,
          cacheSize: 0,
          averageResponseTime: 0,
        });
      }

      return stats;
    } catch (error) {
      throw new StatisticsError(
        `Failed to get statistics by API type: ${error.message}`,
        error
      );
    }
  }

  async getCacheSize(): Promise<number> {
    try {
      const entries = await this.cacheRepository.findAll();

      return entries.reduce((total, entry) => {
        return total + (entry.metadata?.size || 0);
      }, 0);
    } catch (error) {
      throw new StatisticsError(
        `Failed to get cache size: ${error.message}`,
        error
      );
    }
  }

  async getHitRate(): Promise<number> {
    try {
      const stats = await this.getStatistics();
      return stats.hitRate;
    } catch (error) {
      throw new StatisticsError(
        `Failed to get hit rate: ${error.message}`,
        error
      );
    }
  }

  async getMissRate(): Promise<number> {
    try {
      const stats = await this.getStatistics();
      return stats.missRate;
    } catch (error) {
      throw new StatisticsError(
        `Failed to get miss rate: ${error.message}`,
        error
      );
    }
  }

  async getAverageResponseTime(): Promise<number> {
    try {
      const stats = await this.getStatistics();
      return stats.averageResponseTime;
    } catch (error) {
      throw new StatisticsError(
        `Failed to get average response time: ${error.message}`,
        error
      );
    }
  }

  async recordHit(responseTime: number): Promise<void> {
    try {
      const stats = await this.getStatistics();
      const updatedStats = stats.recordHit(responseTime);
      await this.statisticsRepository.save(updatedStats);
    } catch (error) {
      throw new StatisticsError(
        `Failed to record hit: ${error.message}`,
        error
      );
    }
  }

  async recordMiss(responseTime: number): Promise<void> {
    try {
      const stats = await this.getStatistics();
      const updatedStats = stats.recordMiss(responseTime);
      await this.statisticsRepository.save(updatedStats);
    } catch (error) {
      throw new StatisticsError(
        `Failed to record miss: ${error.message}`,
        error
      );
    }
  }

  async updateCacheSize(): Promise<void> {
    try {
      const cacheSize = await this.getCacheSize();
      const stats = await this.getStatistics();
      const updatedStats = stats.updateCacheSize(cacheSize);
      await this.statisticsRepository.save(updatedStats);
    } catch (error) {
      throw new StatisticsError(
        `Failed to update cache size: ${error.message}`,
        error
      );
    }
  }

  async resetStatistics(): Promise<void> {
    try {
      const stats = await this.getStatistics();
      const resetStats = stats.reset();
      await this.statisticsRepository.save(resetStats);
    } catch (error) {
      throw new StatisticsError(
        `Failed to reset statistics: ${error.message}`,
        error
      );
    }
  }
}
```

### 5. DataSourceManagementService

データソースの管理を担当するドメインサービス。

```typescript
export class DataSourceManagementService {
  constructor(
    private readonly dataSourceRepository: DataSourceRepository,
    private readonly cacheInvalidationService: CacheInvalidationService
  ) {}

  async registerDataSource(dataSource: DataSource): Promise<void> {
    try {
      // データソースの重複チェック
      const existing = await this.dataSourceRepository.findById(dataSource.id);
      if (existing) {
        throw new Error(
          `DataSource with ID ${dataSource.id.value} already exists`
        );
      }

      // データソースを保存
      await this.dataSourceRepository.save(dataSource);
    } catch (error) {
      throw new DataSourceError(
        `Failed to register data source: ${error.message}`,
        error
      );
    }
  }

  async updateDataSource(dataSource: DataSource): Promise<void> {
    try {
      // 既存のデータソースを確認
      const existing = await this.dataSourceRepository.findById(dataSource.id);
      if (!existing) {
        throw new Error(`DataSource with ID ${dataSource.id.value} not found`);
      }

      // データソースを更新
      await this.dataSourceRepository.save(dataSource);
    } catch (error) {
      throw new DataSourceError(
        `Failed to update data source: ${error.message}`,
        error
      );
    }
  }

  async activateDataSource(dataSourceId: DataSourceId): Promise<void> {
    try {
      const dataSource = await this.dataSourceRepository.findById(dataSourceId);
      if (!dataSource) {
        throw new Error(`DataSource with ID ${dataSourceId.value} not found`);
      }

      const activatedDataSource = dataSource.activate();
      await this.dataSourceRepository.save(activatedDataSource);
    } catch (error) {
      throw new DataSourceError(
        `Failed to activate data source: ${error.message}`,
        error
      );
    }
  }

  async deactivateDataSource(dataSourceId: DataSourceId): Promise<void> {
    try {
      const dataSource = await this.dataSourceRepository.findById(dataSourceId);
      if (!dataSource) {
        throw new Error(`DataSource with ID ${dataSourceId.value} not found`);
      }

      const deactivatedDataSource = dataSource.deactivate();
      await this.dataSourceRepository.save(deactivatedDataSource);

      // 関連するキャッシュを無効化
      await this.cacheInvalidationService.invalidateByPattern(`estat:*`);
    } catch (error) {
      throw new DataSourceError(
        `Failed to deactivate data source: ${error.message}`,
        error
      );
    }
  }

  async getActiveDataSources(): Promise<DataSource[]> {
    try {
      return await this.dataSourceRepository.findActive();
    } catch (error) {
      throw new DataSourceError(
        `Failed to get active data sources: ${error.message}`,
        error
      );
    }
  }

  async getDataSourceById(
    dataSourceId: DataSourceId
  ): Promise<DataSource | null> {
    try {
      return await this.dataSourceRepository.findById(dataSourceId);
    } catch (error) {
      throw new DataSourceError(
        `Failed to get data source by ID: ${error.message}`,
        error
      );
    }
  }
}
```
