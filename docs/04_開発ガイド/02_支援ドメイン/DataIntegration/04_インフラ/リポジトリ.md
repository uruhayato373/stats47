# DataIntegration ドメイン - リポジトリ

## 概要

DataIntegration ドメインのリポジトリインターフェースと実装を定義します。これらのリポジトリは、外部データソースとの統合とキャッシュ管理のためのデータアクセス層を提供します。

## リポジトリインターフェース

### 1. CacheRepository

R2 ストレージでのキャッシュエントリの管理を担当するリポジトリ。

```typescript
export interface CacheRepository {

  findAll(): Promise<CacheEntry[]>;

  // 統計操作
```

## リポジトリ実装

### 1. R2CacheRepository

Cloudflare R2 ストレージを使用したキャッシュリポジトリの実装。

```typescript
export class R2CacheRepository implements CacheRepository {
  constructor(
    private readonly r2Client: R2Bucket,
    private readonly bucketName: string
  ) {}

  async save(cacheEntry: CacheEntry): Promise<void> {
    try {
      const key = cacheEntry.key.value;
      const data = JSON.stringify(cacheEntry.data);
      const metadata = {
        ttl: cacheEntry.ttl.toString(),
        createdAt: cacheEntry.createdAt.toISOString(),
        lastAccessedAt: cacheEntry.lastAccessedAt.toISOString(),
        hitCount: cacheEntry.hitCount.toString(),
        ...cacheEntry.metadata,
      };

      await this.r2Client.put(key, data, {
        metadata,
        httpMetadata: {
          contentType: "application/json",
          cacheControl: `max-age=${cacheEntry.ttl}`,
        },
      });
    } catch (error) {
      throw new CacheError(
        `Failed to save cache entry: ${error.message}`,
        error
      );
    }
  }

  async findByKey(key: CacheKey): Promise<CacheEntry | null> {
    try {
      const object = await this.r2Client.get(key.value);

      if (!object) {
        return null;
      }

      const data = await object.json();
      const metadata = object.customMetadata || {};

      return CacheEntry.create({
        key,
        data,
        ttl: parseInt(metadata.ttl || "0"),
        metadata: {
          apiType: metadata.apiType || "",
          parameters: JSON.parse(metadata.parameters || "{}"),
          size: parseInt(metadata.size || "0"),
          source: metadata.source || "",
          version: metadata.version || "1.0",
        },
        createdAt: new Date(metadata.createdAt || new Date()),
        lastAccessedAt: new Date(metadata.lastAccessedAt || new Date()),
        hitCount: parseInt(metadata.hitCount || "0"),
      });
    } catch (error) {
      if (error instanceof Error && error.message.includes("NoSuchKey")) {
        return null;
      }
      throw new CacheError(
        `Failed to find cache entry: ${error.message}`,
        error
      );
    }
  }

  async delete(key: CacheKey): Promise<void> {
    try {
      await this.r2Client.delete(key.value);
    } catch (error) {
      throw new CacheError(
        `Failed to delete cache entry: ${error.message}`,
        error
      );
    }
  }

  async findByPattern(pattern: string): Promise<CacheEntry[]> {
    try {
      const listResult = await this.r2Client.list({ prefix: pattern });
      const entries: CacheEntry[] = [];

      for (const object of listResult.objects) {
        const key = CacheKey.create(object.key);
        const entry = await this.findByKey(key);
        if (entry) {
          entries.push(entry);
        }
      }

      return entries;
    } catch (error) {
      throw new CacheError(
        `Failed to find cache entries by pattern: ${error.message}`,
        error
      );
    }
  }

  async findExpired(): Promise<CacheEntry[]> {
    try {
      const allEntries = await this.findAll();
      return allEntries.filter((entry) => entry.isExpired());
    } catch (error) {
      throw new CacheError(
        `Failed to find expired cache entries: ${error.message}`,
        error
      );
    }
  }

  async findOlderThan(date: Date): Promise<CacheEntry[]> {
    try {
      const allEntries = await this.findAll();
      return allEntries.filter((entry) => entry.createdAt < date);
    } catch (error) {
      throw new CacheError(
        `Failed to find old cache entries: ${error.message}`,
        error
      );
    }
  }

  async findByApiType(apiType: ApiType): Promise<CacheEntry[]> {
    const pattern = `estat:${apiType}:`;
    return await this.findByPattern(pattern);
  }

  async deleteByPattern(pattern: string): Promise<number> {
    try {
      const entries = await this.findByPattern(pattern);
      const deletePromises = entries.map((entry) => this.delete(entry.key));
      await Promise.all(deletePromises);
      return entries.length;
    } catch (error) {
      throw new CacheError(
        `Failed to delete cache entries by pattern: ${error.message}`,
        error
      );
    }
  }

  async deleteExpired(): Promise<number> {
    try {
      const expiredEntries = await this.findExpired();
      const deletePromises = expiredEntries.map((entry) =>
        this.delete(entry.key)
      );
      await Promise.all(deletePromises);
      return expiredEntries.length;
    } catch (error) {
      throw new CacheError(
        `Failed to delete expired cache entries: ${error.message}`,
        error
      );
    }
  }

  async deleteAll(): Promise<number> {
    try {
      const allEntries = await this.findAll();
      const deletePromises = allEntries.map((entry) => this.delete(entry.key));
      await Promise.all(deletePromises);
      return allEntries.length;
    } catch (error) {
      throw new CacheError(
        `Failed to delete all cache entries: ${error.message}`,
        error
      );
    }
  }

  async count(): Promise<number> {
    try {
      const listResult = await this.r2Client.list();
      return listResult.objects.length;
    } catch (error) {
      throw new CacheError(
        `Failed to count cache entries: ${error.message}`,
        error
      );
    }
  }

  async getTotalSize(): Promise<number> {
    try {
      const allEntries = await this.findAll();
      return allEntries.reduce(
        (total, entry) => total + (entry.metadata?.size || 0),
        0
      );
    } catch (error) {
      throw new CacheError(
        `Failed to get total cache size: ${error.message}`,
        error
      );
    }
  }

  async getSizeByPattern(pattern: string): Promise<number> {
    try {
      const entries = await this.findByPattern(pattern);
      return entries.reduce(
        (total, entry) => total + (entry.metadata?.size || 0),
        0
      );
    } catch (error) {
      throw new CacheError(
        `Failed to get cache size by pattern: ${error.message}`,
        error
      );
    }
  }

  async findAll(): Promise<CacheEntry[]> {
    try {
      const listResult = await this.r2Client.list();
      const entries: CacheEntry[] = [];

      for (const object of listResult.objects) {
        const key = CacheKey.create(object.key);
        const entry = await this.findByKey(key);
        if (entry) {
          entries.push(entry);
        }
      }

      return entries;
    } catch (error) {
      throw new CacheError(
        `Failed to find all cache entries: ${error.message}`,
        error
      );
    }
  }
}
```

### 2. CloudflareEstatApiRepository

Cloudflare Workers 環境での e-Stat API リポジトリの実装。

```typescript
export class CloudflareEstatApiRepository implements EstatApiRepository {
  constructor(
    private readonly apiKey: string,
    private readonly baseUrl: string = "https://api.e-stat.go.jp/rest/3.0/app/json"
  ) {}

  async getMetaInfo(
    parameters: Record<string, any>
  ): Promise<EstatMetaInfoResponse> {
    const url = this.buildUrl("getMetaInfo", parameters);
    return await this.makeRequest<EstatMetaInfoResponse>(url);
  }

  async getStatsData(
    parameters: Record<string, any>
  ): Promise<EstatStatsDataResponse> {
    const url = this.buildUrl("getStatsData", parameters);
    return await this.makeRequest<EstatStatsDataResponse>(url);
  }

  async getStatsList(
    parameters: Record<string, any>
  ): Promise<EstatStatsListResponse> {
    const url = this.buildUrl("getStatsList", parameters);
    return await this.makeRequest<EstatStatsListResponse>(url);
  }

  async checkRateLimit(): Promise<RateLimitStatus> {
    // e-Stat APIのレート制限チェック
    // 実際の実装では、D1データベースでリクエスト履歴を管理
    return {
      remaining: 1000,
      resetTime: new Date(Date.now() + 24 * 60 * 60 * 1000),
      limit: 1000,
    };
  }

  async waitForRateLimit(): Promise<void> {
    const status = await this.checkRateLimit();
    if (status.remaining <= 0) {
      const waitTime = status.resetTime.getTime() - Date.now();
      if (waitTime > 0) {
        await new Promise((resolve) => setTimeout(resolve, waitTime));
      }
    }
  }

  async handleApiError(error: any): Promise<never> {
    if (error.status === 429) {
      throw new EstatApiError("Rate limit exceeded", error);
    }
    if (error.status === 400) {
      throw new EstatApiError("Invalid request parameters", error);
    }
    if (error.status === 404) {
      throw new EstatApiError("Data not found", error);
    }
    if (error.status >= 500) {
      throw new EstatApiError("Server error", error);
    }
    throw new EstatApiError("Unknown API error", error);
  }

  async retryWithBackoff<T>(
    operation: () => Promise<T>,
    maxRetries: number = 3
  ): Promise<T> {
    let lastError: Error;

    for (let attempt = 0; attempt <= maxRetries; attempt++) {
      try {
        return await operation();
      } catch (error) {
        lastError = error as Error;

        if (attempt === maxRetries) {
          break;
        }

        // 指数バックオフ
        const delay = Math.pow(2, attempt) * 1000;
        await new Promise((resolve) => setTimeout(resolve, delay));
      }
    }

    throw lastError!;
  }

  private buildUrl(apiType: string, parameters: Record<string, any>): string {
    const url = new URL(`${this.baseUrl}/${apiType}`);
    url.searchParams.set("appId", this.apiKey);

    Object.entries(parameters).forEach(([key, value]) => {
      url.searchParams.set(key, String(value));
    });

    return url.toString();
  }

  private async makeRequest<T>(url: string): Promise<T> {
    try {
      const response = await fetch(url);

      if (!response.ok) {
        await this.handleApiError(response);
      }

      const data = await response.json();

      if (data.ERROR_INFO) {
        throw new EstatApiError(`API Error: ${data.ERROR_INFO.MESSAGE}`);
      }

      return data;
    } catch (error) {
      if (error instanceof EstatApiError) {
        throw error;
      }
      throw new EstatApiError(`Request failed: ${error.message}`, error);
    }
  }
}
```

### 3. D1MetadataRepository

Cloudflare D1 データベースを使用したメタデータリポジトリの実装。

```typescript
export class D1MetadataRepository implements MetadataRepository {
  constructor(private readonly d1Client: D1Database) {}

  async saveCacheMetadata(
    cacheKey: CacheKey,
    apiType: ApiType,
    parameters: Record<string, any>
  ): Promise<void> {
    try {
      const expiresAt = new Date(Date.now() + 24 * 60 * 60 * 1000); // 24時間後

      await this.d1Client
        .prepare(
          `
          INSERT OR REPLACE INTO cache_metadata 
          (cache_key, api_type, parameters, expires_at, created_at)
          VALUES (?, ?, ?, ?, ?)
        `
        )
        .bind(
          cacheKey.value,
          apiType,
          JSON.stringify(parameters),
          expiresAt.toISOString(),
          new Date().toISOString()
        )
        .run();
    } catch (error) {
      throw new MetadataError(
        `Failed to save cache metadata: ${error.message}`,
        error
      );
    }
  }

  async findByCacheKey(cacheKey: CacheKey): Promise<CacheMetadata | null> {
    try {
      const result = await this.d1Client
        .prepare("SELECT * FROM cache_metadata WHERE cache_key = ?")
        .bind(cacheKey.value)
        .first();

      if (!result) {
        return null;
      }

      return {
        cacheKey: cacheKey.value,
        apiType: result.api_type,
        parameters: JSON.parse(result.parameters),
        expiresAt: new Date(result.expires_at),
        createdAt: new Date(result.created_at),
      };
    } catch (error) {
      throw new MetadataError(
        `Failed to find cache metadata: ${error.message}`,
        error
      );
    }
  }

  async deleteByCacheKey(cacheKey: CacheKey): Promise<void> {
    try {
      await this.d1Client
        .prepare("DELETE FROM cache_metadata WHERE cache_key = ?")
        .bind(cacheKey.value)
        .run();
    } catch (error) {
      throw new MetadataError(
        `Failed to delete cache metadata: ${error.message}`,
        error
      );
    }
  }

  async deleteByPattern(pattern: string): Promise<number> {
    try {
      const result = await this.d1Client
        .prepare("DELETE FROM cache_metadata WHERE cache_key LIKE ?")
        .bind(`${pattern}%`)
        .run();

      return result.changes;
    } catch (error) {
      throw new MetadataError(
        `Failed to delete cache metadata by pattern: ${error.message}`,
        error
      );
    }
  }

  async deleteExpired(): Promise<number> {
    try {
      const result = await this.d1Client
        .prepare(
          'DELETE FROM cache_metadata WHERE expires_at < datetime("now")'
        )
        .run();

      return result.changes;
    } catch (error) {
      throw new MetadataError(
        `Failed to delete expired cache metadata: ${error.message}`,
        error
      );
    }
  }

  async deleteAll(): Promise<number> {
    try {
      const result = await this.d1Client
        .prepare("DELETE FROM cache_metadata")
        .run();

      return result.changes;
    } catch (error) {
      throw new MetadataError(
        `Failed to delete all cache metadata: ${error.message}`,
        error
      );
    }
  }

  async saveStatistics(statistics: CacheStatistics): Promise<void> {
    try {
      await this.d1Client
        .prepare(
          `
          INSERT OR REPLACE INTO cache_statistics 
          (id, total_requests, hit_count, miss_count, cache_size, average_response_time, last_updated)
          VALUES (?, ?, ?, ?, ?, ?, ?)
        `
        )
        .bind(
          statistics.id.value,
          statistics.totalRequests,
          statistics.hitCount,
          statistics.missCount,
          statistics.cacheSize,
          statistics.averageResponseTime,
          statistics.lastUpdated.toISOString()
        )
        .run();
    } catch (error) {
      throw new MetadataError(
        `Failed to save statistics: ${error.message}`,
        error
      );
    }
  }

  async findLatestStatistics(): Promise<CacheStatistics | null> {
    try {
      const result = await this.d1Client
        .prepare(
          `
          SELECT * FROM cache_statistics 
          ORDER BY last_updated DESC 
          LIMIT 1
        `
        )
        .first();

      if (!result) {
        return null;
      }

      return CacheStatistics.create({
        id: CacheStatisticsId.create(result.id),
        totalRequests: result.total_requests,
        hitCount: result.hit_count,
        missCount: result.miss_count,
        cacheSize: result.cache_size,
        averageResponseTime: result.average_response_time,
        lastUpdated: new Date(result.last_updated),
      });
    } catch (error) {
      throw new MetadataError(
        `Failed to find latest statistics: ${error.message}`,
        error
      );
    }
  }

  async findByApiType(apiType: ApiType): Promise<CacheStatistics | null> {
    try {
      const result = await this.d1Client
        .prepare(
          `
          SELECT * FROM cache_statistics 
          WHERE id LIKE ?
          ORDER BY last_updated DESC 
          LIMIT 1
        `
        )
        .bind(`api_${apiType}%`)
        .first();

      if (!result) {
        return null;
      }

      return CacheStatistics.create({
        id: CacheStatisticsId.create(result.id),
        totalRequests: result.total_requests,
        hitCount: result.hit_count,
        missCount: result.miss_count,
        cacheSize: result.cache_size,
        averageResponseTime: result.average_response_time,
        lastUpdated: new Date(result.last_updated),
      });
    } catch (error) {
      throw new MetadataError(
        `Failed to find statistics by API type: ${error.message}`,
        error
      );
    }
  }

  async updateStatistics(statistics: CacheStatistics): Promise<void> {
    await this.saveStatistics(statistics);
  }

  async deleteStatistics(id: CacheStatisticsId): Promise<void> {
    try {
      await this.d1Client
        .prepare("DELETE FROM cache_statistics WHERE id = ?")
        .bind(id.value)
        .run();
    } catch (error) {
      throw new MetadataError(
        `Failed to delete statistics: ${error.message}`,
        error
      );
    }
  }

  async saveDataSource(dataSource: DataSource): Promise<void> {
    try {
      await this.d1Client
        .prepare(
          `
          INSERT OR REPLACE INTO data_sources 
          (id, name, type, endpoint, authentication, rate_limit, is_active, created_at, updated_at)
          VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
        `
        )
        .bind(
          dataSource.id.value,
          dataSource.name,
          dataSource.type,
          dataSource.endpoint,
          JSON.stringify(dataSource.authentication),
          JSON.stringify(dataSource.rateLimit),
          dataSource.isActive ? 1 : 0,
          dataSource.createdAt.toISOString(),
          dataSource.updatedAt.toISOString()
        )
        .run();
    } catch (error) {
      throw new MetadataError(
        `Failed to save data source: ${error.message}`,
        error
      );
    }
  }

  async findDataSourceById(id: DataSourceId): Promise<DataSource | null> {
    try {
      const result = await this.d1Client
        .prepare("SELECT * FROM data_sources WHERE id = ?")
        .bind(id.value)
        .first();

      if (!result) {
        return null;
      }

      return DataSource.create({
        id: DataSourceId.create(result.id),
        name: result.name,
        type: result.type as DataSourceType,
        endpoint: result.endpoint,
        authentication: JSON.parse(result.authentication),
        rateLimit: JSON.parse(result.rate_limit),
        isActive: Boolean(result.is_active),
        createdAt: new Date(result.created_at),
        updatedAt: new Date(result.updated_at),
      });
    } catch (error) {
      throw new MetadataError(
        `Failed to find data source: ${error.message}`,
        error
      );
    }
  }

  async findActiveDataSources(): Promise<DataSource[]> {
    try {
      const result = await this.d1Client
        .prepare("SELECT * FROM data_sources WHERE is_active = 1")
        .all();

      return result.results.map((row) =>
        DataSource.create({
          id: DataSourceId.create(row.id),
          name: row.name,
          type: row.type as DataSourceType,
          endpoint: row.endpoint,
          authentication: JSON.parse(row.authentication),
          rateLimit: JSON.parse(row.rate_limit),
          isActive: Boolean(row.is_active),
          createdAt: new Date(row.created_at),
          updatedAt: new Date(row.updated_at),
        })
      );
    } catch (error) {
      throw new MetadataError(
        `Failed to find active data sources: ${error.message}`,
        error
      );
    }
  }

  async updateDataSource(dataSource: DataSource): Promise<void> {
    await this.saveDataSource(dataSource);
  }

  async deleteDataSource(id: DataSourceId): Promise<void> {
    try {
      await this.d1Client
        .prepare("DELETE FROM data_sources WHERE id = ?")
        .bind(id.value)
        .run();
    } catch (error) {
      throw new MetadataError(
        `Failed to delete data source: ${error.message}`,
        error
      );
    }
  }
}
```
