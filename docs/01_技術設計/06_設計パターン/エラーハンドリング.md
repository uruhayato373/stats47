# エラーハンドリング

## 概要

stats47 プロジェクトにおけるエラーハンドリングの統一戦略について説明します。

## エラーハンドリング戦略

### 基本原則

1. **統一性**: 全システムで一貫したエラー処理
2. **ユーザビリティ**: ユーザーに分かりやすいエラーメッセージ
3. **デバッグ性**: 開発者にとって有用なエラー情報
4. **復旧性**: エラーからの自動復旧機能

### エラーの分類

```typescript
// src/types/error.ts

  NETWORK = "NETWORK",

  message: string;
  code?: string;
  details?: unknown;
```

## エラーハンドリングパターン

### 1. API エラーハンドリング

#### 統一エラーレスポンス

```typescript
// src/lib/api/error-handler.ts
export class ApiError extends Error {
  constructor(
    public status: number,
    public code: string,
    message: string,
    public details?: unknown
  ) {
    super(message);
    this.name = "ApiError";
  }
}

export function handleApiError(error: unknown): AppError {
  if (error instanceof ApiError) {
    return {
      type: getErrorType(error.status),
      message: error.message,
      code: error.code,
      details: error.details,
      timestamp: new Date(),
    };
  }

  if (error instanceof Error) {
    return {
      type: ErrorType.CLIENT,
      message: error.message,
      timestamp: new Date(),
      stack: error.stack,
    };
  }

  return {
    type: ErrorType.CLIENT,
    message: "Unknown error occurred",
    timestamp: new Date(),
  };
}

function getErrorType(status: number): ErrorType {
  switch (status) {
    case 400:
      return ErrorType.VALIDATION;
    case 401:
      return ErrorType.AUTHENTICATION;
    case 403:
      return ErrorType.AUTHORIZATION;
    case 404:
      return ErrorType.NOT_FOUND;
    case 500:
    case 502:
    case 503:
      return ErrorType.SERVER;
    default:
      return ErrorType.CLIENT;
  }
}
```

#### API クライアントでのエラーハンドリング

```typescript
// src/lib/api/client.ts
export class ApiClient {
  async request<T>(url: string, options?: RequestInit): Promise<T> {
    try {
      const response = await fetch(url, {
        ...options,
        headers: {
          "Content-Type": "application/json",
          ...options?.headers,
        },
      });

      if (!response.ok) {
        const errorData = await response.json().catch(() => ({}));
        throw new ApiError(
          response.status,
          errorData.code || "UNKNOWN_ERROR",
          errorData.message || "Request failed",
          errorData.details
        );
      }

      return await response.json();
    } catch (error) {
      if (error instanceof ApiError) {
        throw error;
      }

      // ネットワークエラー
      throw new ApiError(0, "NETWORK_ERROR", "Network request failed", error);
    }
  }
}
```

### 2. React コンポーネントでのエラーハンドリング

#### Error Boundary

```typescript
// src/components/ErrorBoundary.tsx
"use client";

import React from "react";

interface ErrorBoundaryState {
  hasError: boolean;
  error?: Error;
}

export class ErrorBoundary extends React.Component<
  React.PropsWithChildren<{}>,
  ErrorBoundaryState
> {
  constructor(props: React.PropsWithChildren<{}>) {
    super(props);
    this.state = { hasError: false };
  }

  static getDerivedStateFromError(error: Error): ErrorBoundaryState {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, errorInfo: React.ErrorInfo) {
    console.error("ErrorBoundary caught an error:", error, errorInfo);

    // エラーログの送信
    this.logError(error, errorInfo);
  }

  private logError(error: Error, errorInfo: React.ErrorInfo) {
    // エラーログの送信ロジック
    fetch("/api/errors", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        message: error.message,
        stack: error.stack,
        componentStack: errorInfo.componentStack,
        timestamp: new Date().toISOString(),
      }),
    });
  }

  render() {
    if (this.state.hasError) {
      return <ErrorFallback error={this.state.error} />;
    }

    return this.props.children;
  }
}

function ErrorFallback({ error }: { error?: Error }) {
  return (
    <div className="error-fallback">
      <h2>Something went wrong</h2>
      <p>{error?.message || "An unexpected error occurred"}</p>
      <button onClick={() => window.location.reload()}>Reload Page</button>
    </div>
  );
}
```

#### SWR でのエラーハンドリング

```typescript
// src/hooks/useApiData.ts
import useSWR from "swr";
import { handleApiError } from "@/lib/api/error-handler";

export function useApiData<T>(url: string) {
  const { data, error, mutate } = useSWR<T, AppError>(
    url,
    async (url: string) => {
      try {
        const response = await fetch(url);
        if (!response.ok) {
          throw new Error(`HTTP ${response.status}`);
        }
        return await response.json();
      } catch (error) {
        throw handleApiError(error);
      }
    },
    {
      onError: (error) => {
        console.error("SWR Error:", error);
        // エラーログの送信
        logError(error);
      },
      errorRetryCount: 3,
      errorRetryInterval: 5000,
    }
  );

  return {
    data,
    error,
    isLoading: !error && !data,
    mutate,
  };
}
```

### 3. フォームバリデーション

#### バリデーションエラーハンドリング

```typescript
// src/lib/validation/error-handler.ts
export interface ValidationError {
  field: string;
  message: string;
  code: string;
}

export class ValidationErrorHandler {
  static handleValidationError(error: unknown): ValidationError[] {
    if (Array.isArray(error)) {
      return error.map((err) => ({
        field: err.field || "unknown",
        message: err.message || "Validation failed",
        code: err.code || "VALIDATION_ERROR",
      }));
    }

    if (error instanceof Error) {
      return [
        {
          field: "general",
          message: error.message,
          code: "VALIDATION_ERROR",
        },
      ];
    }

    return [
      {
        field: "general",
        message: "Validation failed",
        code: "VALIDATION_ERROR",
      },
    ];
  }
}
```

#### フォームコンポーネントでのエラー表示

```typescript
// src/components/Form.tsx
"use client";

import { useState } from "react";
import { ValidationErrorHandler } from "@/lib/validation/error-handler";

export function Form() {
  const [errors, setErrors] = useState<ValidationError[]>([]);
  const [isSubmitting, setIsSubmitting] = useState(false);

  const handleSubmit = async (data: FormData) => {
    try {
      setIsSubmitting(true);
      setErrors([]);

      await submitForm(data);
    } catch (error) {
      const validationErrors =
        ValidationErrorHandler.handleValidationError(error);
      setErrors(validationErrors);
    } finally {
      setIsSubmitting(false);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      {errors.map((error, index) => (
        <div key={index} className="error-message">
          {error.message}
        </div>
      ))}

      <button type="submit" disabled={isSubmitting}>
        {isSubmitting ? "Submitting..." : "Submit"}
      </button>
    </form>
  );
}
```

## エラーログ・監視

### 1. エラーログの送信

```typescript
// src/lib/error-logger.ts
export class ErrorLogger {
  static async logError(error: AppError, context?: unknown) {
    try {
      await fetch("/api/errors", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          ...error,
          context,
          userAgent: navigator.userAgent,
          url: window.location.href,
          timestamp: new Date().toISOString(),
        }),
      });
    } catch (logError) {
      console.error("Failed to log error:", logError);
    }
  }

  static logToConsole(error: AppError, context?: unknown) {
    console.error("Application Error:", {
      ...error,
      context,
    });
  }
}
```

### 2. エラー監視

```typescript
// src/lib/error-monitor.ts
export class ErrorMonitor {
  private static instance: ErrorMonitor;
  private errorCount = 0;
  private maxErrors = 10;

  static getInstance(): ErrorMonitor {
    if (!ErrorMonitor.instance) {
      ErrorMonitor.instance = new ErrorMonitor();
    }
    return ErrorMonitor.instance;
  }

  reportError(error: AppError) {
    this.errorCount++;

    if (this.errorCount > this.maxErrors) {
      console.warn("Too many errors reported, stopping error reporting");
      return;
    }

    ErrorLogger.logError(error);
  }

  reset() {
    this.errorCount = 0;
  }
}
```

### 1. エラーメッセージの国際化

```typescript
// src/lib/error-messages.ts
export const ERROR_MESSAGES = {
  [ErrorType.NETWORK]: {
    ja: "ネットワークエラーが発生しました。インターネット接続を確認してください。",
    en: "Network error occurred. Please check your internet connection.",
  },
  [ErrorType.VALIDATION]: {
    ja: "入力内容に問題があります。確認してください。",
    en: "There is a problem with your input. Please check.",
  },
  [ErrorType.AUTHENTICATION]: {
    ja: "認証に失敗しました。ログインし直してください。",
    en: "Authentication failed. Please log in again.",
  },
  [ErrorType.NOT_FOUND]: {
    ja: "お探しのページが見つかりません。",
    en: "The page you are looking for was not found.",
  },
  [ErrorType.SERVER]: {
    ja: "サーバーエラーが発生しました。しばらくしてから再試行してください。",
    en: "Server error occurred. Please try again later.",
  },
};

export function getErrorMessage(
  error: AppError,
  locale: string = "ja"
): string {
  const messages = ERROR_MESSAGES[error.type];
  return messages[locale as keyof typeof messages] || messages.ja;
}
```

### 2. エラー表示コンポーネント

```typescript
// src/components/ErrorMessage.tsx
interface ErrorMessageProps {
  error: AppError;
  locale?: string;
  onRetry?: () => void;

export function ErrorMessage({
  error,
  locale = "ja",
  onRetry,

          </button>

  );
}
```

```typescript
// src/lib/__tests__/error-handler.test.ts
import { handleApiError, ApiError } from "../error-handler";

describe("Error Handler", () => {
  it("should handle API errors correctly", () => {
    const apiError = new ApiError(404, "NOT_FOUND", "Resource not found");
    const result = handleApiError(apiError);

    expect(result.type).toBe("NOT_FOUND");
    expect(result.message).toBe("Resource not found");
    expect(result.code).toBe("NOT_FOUND");
  });

  it("should handle network errors", () => {
    const networkError = new Error("Network request failed");
    const result = handleApiError(networkError);

    expect(result.type).toBe("CLIENT");
    expect(result.message).toBe("Network request failed");
  });
});
```
