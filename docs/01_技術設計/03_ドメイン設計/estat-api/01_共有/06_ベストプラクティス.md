# コード品質

## 1. 型安全性

### 良い例: 明確な型定義

```typescript
interface StatsDataOptions {
  categoryFilter?: string;
  yearFilter?: string;
  areaFilter?: string;
  limit?: number;
}

async function getStatsData(
  statsDataId: string,
  options: StatsDataOptions
): Promise<FormattedStatsData> {
  return await EstatStatsDataService.getAndFormatStatsData(
    statsDataId,
    options
  );
}
```

### 悪い例: any型の使用

```typescript
// ❌ 型安全性が失われる
function getStatsData(options: any) {
  // 型チェックが効かない
}
```

## 2. エラーハンドリング

### 良い例: 適切なエラーハンドリング

```typescript
try {
  const data = await EstatStatsDataService.getAndFormatStatsData(statsDataId);
  return data;
} catch (error) {
  if (error instanceof EstatStatsDataFetchError) {
    console.error("データ取得エラー:", error.message);
    throw error;
  }
  throw new EstatStatsDataFetchError(
    "予期しないエラーが発生しました",
    statsDataId,
    error
  );
}
```

### 悪い例: エラーの無視

```typescript
// ❌ エラーが発生しても処理が続行される
const data = await EstatStatsDataService.getAndFormatStatsData(statsDataId);
```

## 3. 非同期処理

### 良い例: Promise.allSettledを使用

```typescript
const results = await Promise.allSettled(
  statsDataIds.map((id) => EstatStatsDataService.getAndFormatStatsData(id))
);

const successful = results
  .filter(
    (result): result is PromiseFulfilledResult<FormattedStatsData> =>
      result.status === "fulfilled"
  )
  .map((result) => result.value);
```

### 悪い例: 順次処理

```typescript
// ❌ 非効率な順次処理
for (const id of statsDataIds) {
  const data = await EstatStatsDataService.getAndFormatStatsData(id);
}
```

## 2. キャッシュの活用

```typescript
class CachedDataFetcher {
  private cache = new Map<string, { data: any; timestamp: number }>();
  private ttl = 5 * 60 * 1000; // 5分

  async fetchWithCache(statsDataId: string, options = {}) {
    const cacheKey = `${statsDataId}-${JSON.stringify(options)}`;
    const cached = this.cache.get(cacheKey);

    if (cached && Date.now() - cached.timestamp < this.ttl) {
      return cached.data;
    }

    const data = await EstatStatsDataService.getAndFormatStatsData(
      statsDataId,
      options
    );
    this.cache.set(cacheKey, { data, timestamp: Date.now() });

    return data;
  }
}
```

## 3. バッチ処理

```typescript
const processMultipleStatsData = async (statsDataIds: string[]) => {
  const batchSize = 10;
  const batches = chunk(statsDataIds, batchSize);

  const results = [];
  for (const batch of batches) {
    const batchResults = await Promise.allSettled(
      batch.map((id) => EstatStatsDataService.getAndFormatStatsData(id))
    );
    results.push(...batchResults);

    // レート制限を考慮した待機
    await new Promise((resolve) => setTimeout(resolve, 1000));
  }

  return results;
};
```

## 4. データ圧縮

```typescript
// 大量のデータを扱う場合は圧縮を検討
import { gzip, gunzip } from "zlib";
import { promisify } from "util";

const gzipAsync = promisify(gzip);
const gunzipAsync = promisify(gunzip);

async function compressData(data: any): Promise<Buffer> {
  const json = JSON.stringify(data);
  return await gzipAsync(Buffer.from(json));
}

async function decompressData(compressed: Buffer): Promise<any> {
  const decompressed = await gunzipAsync(compressed);
  return JSON.parse(decompressed.toString());
}
```

## 5. リトライ戦略

```typescript
async function fetchWithExponentialBackoff(
  statsDataId: string,
  maxRetries: number = 3
) {
  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      return await EstatStatsDataService.getAndFormatStatsData(statsDataId);
    } catch (error) {
      if (attempt === maxRetries) throw error;

      const delay = Math.min(1000 * Math.pow(2, attempt - 1), 10000);
      await new Promise((resolve) => setTimeout(resolve, delay));
    }
  }
}
```

# セキュリティ

## 1. 入力検証

### zodを使用した検証

```typescript
import { z } from "zod";

const StatsDataIdSchema = z.string().regex(/^\d{10}$/, {
  message: "統計表IDは10桁の数字である必要があります",
});

const StatsDataOptionsSchema = z.object({
  categoryFilter: z.string().optional(),
  yearFilter: z.string().regex(/^\d{4}$/).optional(),
  areaFilter: z.string().optional(),
  limit: z.number().min(1).max(10000).optional(),
});

async function getStatsData(statsDataId: string, options: unknown) {
  // 検証
  const validatedId = StatsDataIdSchema.parse(statsDataId);
  const validatedOptions = StatsDataOptionsSchema.parse(options);

  return await EstatStatsDataService.getAndFormatStatsData(
    validatedId,
    validatedOptions
  );
}
```

## 3. レート制限

```typescript
class RateLimiter {
  private requests: number[] = [];
  private readonly maxRequests: number;
  private readonly windowMs: number;

  constructor(maxRequests: number, windowMs: number) {
    this.maxRequests = maxRequests;
    this.windowMs = windowMs;
  }

  async checkLimit(): Promise<void> {
    const now = Date.now();
    this.requests = this.requests.filter((time) => now - time < this.windowMs);

    if (this.requests.length >= this.maxRequests) {
      const waitTime = this.windowMs - (now - this.requests[0]);
      await new Promise((resolve) => setTimeout(resolve, waitTime));
    }

    this.requests.push(now);
  }
}
```

async function measurePerformance<T>(
  name: string,
  operation: () => Promise<T>
): Promise<T> {
  const start = performance.now();

  try {
    const result = await operation();
    const duration = performance.now() - start;
    console.log(`${name}: ${duration.toFixed(2)}ms`);
    return result;
  } catch (error) {
    const duration = performance.now() - start;
    console.error(`${name} failed after ${duration.toFixed(2)}ms`);
    throw error;
  }
}

## 4. 監視とメトリクス

```typescript
class MetricsCollector {
  private metrics = new Map<string, number[]>();

  recordMetric(name: string, value: number): void {
    if (!this.metrics.has(name)) {
      this.metrics.set(name, []);
    }
    this.metrics.get(name)!.push(value);
  }

  getMetrics(name: string) {
    const values = this.metrics.get(name) || [];
    return {
      count: values.length,
      avg: values.reduce((a, b) => a + b, 0) / values.length,
      min: Math.min(...values),
      max: Math.max(...values),
    };
  }
}
```
