# 状態管理設計

## 概要

stats47 プロジェクトでは、Jotai を基盤とした状態管理アーキテクチャを採用しています。

## 状態管理アーキテクチャ

### 基本原則

1. **Atomic Design**: 状態を最小単位に分割
2. **単方向データフロー**: データの流れを明確化
3. **型安全性**: TypeScript による型チェック
4. **パフォーマンス**: 必要な部分のみ再レンダリング

### 状態の分類

```typescript
// 1. プリミティブ状態
export const themeAtom = atomWithStorage<Theme>("theme", "light");
export const selectedCategoryAtom = atom<string | null>(null);

// 2. 派生状態
export const effectiveThemeAtom = atom((get) => {
  const theme = get(themeAtom);
  const mounted = get(mountedAtom);
  return mounted ? theme : "light";
});

// 3. 非同期状態
export const dataAtom = atom(async () => {
  const response = await fetch("/api/data");
  return response.json();
});

// 4. 書き込み専用状態（アクション）
export const toggleThemeAtom = atom(null, (get, set) => {
  const current = get(themeAtom);
  set(themeAtom, current === "light" ? "dark" : "light");
});
```

## 状態管理パターン

### 1. データフェッチパターン

```typescript
// src/store/data-fetch.ts
export const dataAtom = atom(async () => {
  const response = await fetch("/api/data");
  if (!response.ok) {
    throw new Error("Failed to fetch data");
  }
  return response.json();
});

export const dataWithErrorHandlingAtom = atom(async (get) => {
  try {
    return await get(dataAtom);
  } catch (error) {
    console.error("Data fetch error:", error);
    return null;
  }
});
```

### 2. フォーム状態管理

```typescript
// src/store/form.ts
export const formDataAtom = atom<FormData>({});
export const formErrorsAtom = atom<FormErrors>({});
export const formTouchedAtom = atom<FormTouched>({});

export const updateFormFieldAtom = atom(
```

## パフォーマンス最適化

### 1. アトムの分割

```typescript
// ❌ 悪い例: 大きなオブジェクトを一つのアトムで管理
export const largeStateAtom = atom<LargeState>({
  user: { ... },
  data: { ... },
  ui: { ... }
});

// ✅ 良い例: 小さな単位に分割
export const userAtom = atom<User>({ ... });
export const dataAtom = atom<Data>({ ... });
export const uiAtom = atom<UI>({ ... });
```

### 2. メモ化の活用

```typescript
// src/store/memoized.ts
export const expensiveComputationAtom = atom((get) => {
```

## テスト戦略

### 1. アトムのテスト

```typescript
// src/store/__tests__/theme.test.ts
import { renderHook } from "@testing-library/react";
import { useAtom } from "jotai";
import { themeAtom, toggleThemeAtom } from "../theme";

describe("Theme Atoms", () => {
  it("should toggle theme correctly", () => {
    const { result } = renderHook(() => useAtom(themeAtom));
    const [, setTheme] = result.current;

    // 初期状態はlight
    expect(result.current[0]).toBe("light");

    // toggleを実行
    const { result: toggleResult } = renderHook(() => useAtom(toggleThemeAtom));
    const [, toggle] = toggleResult.current;
    toggle();

    // darkに変更される
    expect(result.current[0]).toBe("dark");
  });
});
```

### 2. 統合テスト

```typescript
// src/store/__tests__/integration.test.ts
import { renderHook } from "@testing-library/react";
import { Provider } from "jotai";
import { useAtom } from "jotai";

describe("State Management Integration", () => {
  it("should work with multiple atoms", () => {
    const wrapper = ({ children }: { children: React.ReactNode }) => (
      <Provider>{children}</Provider>
    );

    const { result } = renderHook(
      () => {
        const [theme] = useAtom(themeAtom);
        const [user] = useAtom(userAtom);
        return { theme, user };
      },
      { wrapper }
    );

    expect(result.current.theme).toBe("light");
    expect(result.current.user).toBeNull();
  });
});
```
