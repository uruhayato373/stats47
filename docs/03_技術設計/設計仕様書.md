### 6.1 ランキング計算

```typescript
// src/lib/utils/ranking.ts

import { PrefectureRanking } from "@/lib/types/prefecture";

export function calculateRanking(
  data: Array<{ code: string; name: string; value: number }>,
  options?: {
    sortDirection?: "asc" | "desc";
    calculatePercentile?: boolean;
    calculateDeviation?: boolean;
  }
): PrefectureRanking[] {
  const {
    sortDirection = "desc",
    calculatePercentile = true,
    calculateDeviation = true,
  } = options || {};

  // ソート
  const sorted = [...data].sort((a, b) =>
    sortDirection === "desc" ? b.value - a.value : a.value - b.value
  );

  // 統計値計算
  const values = sorted.map((d) => d.value);
  const mean = values.reduce((sum, v) => sum + v, 0) / values.length;
  const variance =
    values.reduce((sum, v) => sum + Math.pow(v - mean, 2), 0) / values.length;
  const stdDev = Math.sqrt(variance);

  return sorted.map((item, index) => {
    const rank = index + 1;
    const percentile = calculatePercentile
      ? ((sorted.length - rank) / sorted.length) * 100
      : 0;
    const deviation = calculateDeviation
      ? 50 + ((item.value - mean) / stdDev) * 10
      : 50;

    return {
      prefectureCode: item.code,
      prefectureName: item.name,
      value: item.value,
      rank,
      percentile: Math.round(percentile * 10) / 10,
      deviation: Math.round(deviation * 10) / 10,
    };
  });
}
```

### 6.2 色スケール計算

```typescript
// src/lib/utils/colorScale.ts

import { scaleQuantile, scaleSequential } from "d3-scale";
import { interpolateYlOrRd, interpolateBlues } from "d3-scale-chromatic";

export function createColorScale(
  values: number[],
  colorScheme: "sequential" | "diverging" = "sequential"
): (value: number) => string {
  const validValues = values.filter((v) => v !== null && !isNaN(v));

  if (colorScheme === "sequential") {
    const scale = scaleSequential(interpolateYlOrRd).domain([
      Math.min(...validValues),
      Math.max(...validValues),
    ]);
    return (value: number) => scale(value);
  }

  // Quantile scale（等頻度区分）
  const scale = scaleQuantile<string>()
    .domain(validValues)
    .range([
      "#f7fbff",
      "#deebf7",
      "#c6dbef",
      "#9ecae1",
      "#6baed6",
      "#4292c6",
      "#2171b5",
      "#084594",
    ]);

  return (value: number) => scale(value);
}

export function getLegendSteps(
  values: number[],
  steps: number = 5
): Array<{ min: number; max: number; color: string }> {
  const validValues = values.filter((v) => v !== null && !isNaN(v));
  const min = Math.min(...validValues);
  const max = Math.max(...validValues);
  const step = (max - min) / steps;

  const colorScale = createColorScale(values);

  return Array.from({ length: steps }, (_, i) => ({
    min: min + step * i,
    max: min + step * (i + 1),
    color: colorScale(min + step * i + step / 2),
  }));
}
```

// src/lib/utils/format.ts

import { Indicator, IndicatorFormat } from "@/lib/types/indicator";

export function formatValue(
  value: number | null,
  indicator: Indicator
): string {
  if (value === null || isNaN(value)) return "-";

  const decimals = indicator.decimalPlaces ?? 0;

  switch (indicator.format) {
    case "percentage":
      return `${value.toFixed(decimals)}%`;
    case "currency":
      return `${value.toLocaleString("ja-JP")}${indicator.unit}`;
    case "rate":
      return `${value.toFixed(decimals)}${indicator.unit}`;
    case "number":
    default:
      return `${value.toLocaleString("ja-JP", {
        minimumFractionDigits: decimals,
        maximumFractionDigits: decimals,
      })}${indicator.unit}`;
  }
}

export function formatRank(rank: number): string {
  return `${rank}位`;
}

export function formatChange(change: number): { text: string; color: string } {
  if (change === 0) return { text: "→", color: "text-gray-500" };
  if (change > 0)
    return { text: `↑${Math.abs(change)}`, color: "text-green-600" };
  return { text: `↓${Math.abs(change)}`, color: "text-red-600" };
}

  },
};

// Next.js fetch with caching
export async function fetchWithCache<T>(
  url: string,
  options?: RequestInit & { next?: { revalidate?: number | false } }
): Promise<T> {
  const response = await fetch(url, {
    ...options,
    next: {
      revalidate: options?.next?.revalidate ?? 3600,
    },
  });

  if (!response.ok) {
    throw new Error(`Failed to fetch: ${url}`);
  }

  return response.json();
}
