# 都道府県ランキング可視化アプリケーション 設計仕様書

## 1. プロジェクト概要

### 目的

- 都道府県別の統計データをコロプレス地図で可視化
- ランキング形式でのデータ比較
- 市町村レベルの詳細データ表示
- インタラクティブなダッシュボード提供

### 主要機能

1. 都道府県コロプレス地図（色分けマップ）
2. 都道府県ランキング表示
3. 都道府県詳細ダッシュボード
4. 市町村ランキング表示
5. データフィルタリング・検索

## 2. 技術スタック

```yaml
Framework: Next.js 14 (App Router)
Language: TypeScript
Styling: Tailwind CSS
```

```typescript
import { useEffect, useState } from "react";
import { Prefecture, PrefectureIndicator } from "@/lib/types/prefecture";

export function usePrefectureData(year?: number) {
  const [prefectures, setPrefectures] = useState<Prefecture[]>([]);
  const [indicators, setIndicators] = useState<PrefectureIndicator[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<Error | null>(null);

  useEffect(() => {
    async function fetchData() {
      try {
        setLoading(true);
        const response = await fetch(
          `/api/prefectures${year ? `?year=${year}` : ""}`
        );
        const data = await response.json();
        setPrefectures(data.prefectures);
        setIndicators(data.indicators);
      } catch (err) {
        setError(err as Error);
      } finally {
        setLoading(false);
      }
    }
    fetchData();
  }, [year]);

  return { prefectures, indicators, loading, error };
}
```

## 6. ユーティリティ関数

### 6.1 ランキング計算

```typescript
// src/lib/utils/ranking.ts

import { PrefectureRanking } from "@/lib/types/prefecture";

export function calculateRanking(
  data: Array<{ code: string; name: string; value: number }>,
  options?: {
    sortDirection?: "asc" | "desc";
    calculatePercentile?: boolean;
    calculateDeviation?: boolean;
  }
): PrefectureRanking[] {
  const {
    sortDirection = "desc",
    calculatePercentile = true,
    calculateDeviation = true,
  } = options || {};

  // ソート
  const sorted = [...data].sort((a, b) =>
    sortDirection === "desc" ? b.value - a.value : a.value - b.value
  );

  // 統計値計算
  const values = sorted.map((d) => d.value);
  const mean = values.reduce((sum, v) => sum + v, 0) / values.length;
  const variance =
    values.reduce((sum, v) => sum + Math.pow(v - mean, 2), 0) / values.length;
  const stdDev = Math.sqrt(variance);

  return sorted.map((item, index) => {
    const rank = index + 1;
    const percentile = calculatePercentile
      ? ((sorted.length - rank) / sorted.length) * 100
      : 0;
    const deviation = calculateDeviation
      ? 50 + ((item.value - mean) / stdDev) * 10
      : 50;

    return {
      prefectureCode: item.code,
      prefectureName: item.name,
      value: item.value,
      rank,
      percentile: Math.round(percentile * 10) / 10,
      deviation: Math.round(deviation * 10) / 10,
    };
  });
}
```

### 6.2 色スケール計算

```typescript
// src/lib/utils/colorScale.ts

import { scaleQuantile, scaleSequential } from "d3-scale";
import { interpolateYlOrRd, interpolateBlues } from "d3-scale-chromatic";

export function createColorScale(
  values: number[],
  colorScheme: "sequential" | "diverging" = "sequential"
): (value: number) => string {
  const validValues = values.filter((v) => v !== null && !isNaN(v));

  if (colorScheme === "sequential") {
    const scale = scaleSequential(interpolateYlOrRd).domain([
      Math.min(...validValues),
      Math.max(...validValues),
    ]);
    return (value: number) => scale(value);
  }

  // Quantile scale（等頻度区分）
  const scale = scaleQuantile<string>()
    .domain(validValues)
    .range([
      "#f7fbff",
      "#deebf7",
      "#c6dbef",
      "#9ecae1",
      "#6baed6",
      "#4292c6",
      "#2171b5",
      "#084594",
    ]);

  return (value: number) => scale(value);
}

export function getLegendSteps(
  values: number[],
  steps: number = 5
): Array<{ min: number; max: number; color: string }> {
  const validValues = values.filter((v) => v !== null && !isNaN(v));
  const min = Math.min(...validValues);
  const max = Math.max(...validValues);
  const step = (max - min) / steps;

  const colorScale = createColorScale(values);

  return Array.from({ length: steps }, (_, i) => ({
    min: min + step * i,
    max: min + step * (i + 1),
    color: colorScale(min + step * i + step / 2),
  }));
}
```

```typescript
// src/lib/utils/format.ts

import { Indicator, IndicatorFormat } from "@/lib/types/indicator";

export function formatValue(
  value: number | null,
  indicator: Indicator
): string {
  if (value === null || isNaN(value)) return "-";

  const decimals = indicator.decimalPlaces ?? 0;

  switch (indicator.format) {
    case "percentage":
      return `${value.toFixed(decimals)}%`;
    case "currency":
      return `${value.toLocaleString("ja-JP")}${indicator.unit}`;
    case "rate":
      return `${value.toFixed(decimals)}${indicator.unit}`;
    case "number":
    default:
      return `${value.toLocaleString("ja-JP", {
        minimumFractionDigits: decimals,
        maximumFractionDigits: decimals,
      })}${indicator.unit}`;
  }
}

export function formatRank(rank: number): string {
  return `${rank}位`;
}

export function formatChange(change: number): { text: string; color: string } {
  if (change === 0) return { text: "→", color: "text-gray-500" };
  if (change > 0)
    return { text: `↑${Math.abs(change)}`, color: "text-green-600" };
  return { text: `↓${Math.abs(change)}`, color: "text-red-600" };
}
```

### 7.2 コロプレス地図（概要）

```typescript
// src/components/maps/ChoroplethMap.tsx

"use client";

import { useEffect, useRef } from "react";
import * as d3 from "d3";
import {
  GeoJSONFeatureCollection,
  PrefectureGeoProperties,
} from "@/lib/types/geo";
import { Indicator } from "@/lib/types/indicator";
import { createColorScale } from "@/lib/utils/colorScale";

interface Props {
  geojson: GeoJSONFeatureCollection<PrefectureGeoProperties>;
  data: Map<string, number>;
  indicator: Indicator;
  onRegionClick?: (code: string) => void;
  selectedCode?: string;
}

export function ChoroplethMap({
  geojson,
  data,
  indicator,
  onRegionClick,
  selectedCode,
}: Props) {
  const svgRef = useRef<SVGSVGElement>(null);

  useEffect(() => {
    if (!svgRef.current) return;

    const svg = d3.select(svgRef.current);
    const width = 800;
    const height = 600;

    // 投影法の設定
    const projection = d3.geoMercator().fitSize([width, height], geojson);
    const path = d3.geoPath().projection(projection);

    // 色スケールの作成
    const values = Array.from(data.values());
    const colorScale = createColorScale(values);

    // 地図の描画
    svg
      .selectAll("path")
      .data(geojson.features)
      .join("path")
      .attr("d", path)
      .attr("fill", (d) => {
        const value = data.get(d.properties.prefCode);
        return value !== undefined ? colorScale(value) : "#cccccc";
      })
      .attr("stroke", (d) =>
        d.properties.prefCode === selectedCode ? "#000000" : "#ffffff"
      )
      .attr("stroke-width", (d) =>
        d.properties.prefCode === selectedCode ? 2 : 0.5
      )
      .style("cursor", "pointer")
      .on("click", (event, d) => {
        onRegionClick?.(d.properties.prefCode);
      })
      .on("mouseover", function () {
        d3.select(this).attr("opacity", 0.8);
      })
      .on("mouseout", function () {
        d3.select(this).attr("opacity", 1);
      });
  }, [geojson, data, indicator, onRegionClick, selectedCode]);

  return (
    <div className="relative">
      <svg
        ref={svgRef}
        width={800}
        height={600}
        className="border border-gray-200 rounded"
      />
    </div>
  );
}
```

```typescript
// Next.js fetch with caching
export async function fetchWithCache<T>(
  url: string,
  options?: RequestInit & { next?: { revalidate?: number | false } }
): Promise<T> {
  const response = await fetch(url, {
    ...options,
    next: {
      revalidate: options?.next?.revalidate ?? 3600,
    },
  });

  if (!response.ok) {
    throw new Error(`Failed to fetch: ${url}`);
  }

  return response.json();
}
```
